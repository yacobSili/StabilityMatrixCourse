# 页面回收框架详解

## 学习目标

- 理解内存回收的触发条件和时机
- 掌握 LRU 链表的组织和管理
- 了解 kswapd 后台回收机制
- 理解内存水位线（watermark）的作用

## 一、内存回收概述

### 1.1 为什么需要内存回收

- 物理内存有限，需要回收不活跃页面
- 保持足够空闲内存应对突发分配
- 平衡文件缓存和应用程序内存

### 1.2 回收触发条件

```
                    空闲内存
                        │
    high_wmark ─────────┼───────────────────────
                        │   正常分配
                        │
    low_wmark  ─────────┼───────────────────────
                        │   唤醒 kswapd
                        │
    min_wmark  ─────────┼───────────────────────
                        │   触发直接回收
                        │   高优先级分配
                        │
        0 ──────────────┼───────────────────────
```

| 触发点 | 条件 | 回收方式 |
|-------|------|---------|
| kswapd 唤醒 | 空闲内存 < low_wmark | 后台异步回收 |
| 直接回收 | 空闲内存 < min_wmark | 同步回收（阻塞） |
| 分配失败 | 回收后仍不足 | OOM Killer |

### 1.3 可回收页面类型

```c
// 可回收的页面类型
// 1. 文件页（Page Cache）
//    - 干净页：直接丢弃
//    - 脏页：回写后丢弃

// 2. 匿名页
//    - 需要换出到 swap
//    - 或者 swap 满则不能回收

// 3. Slab 缓存
//    - 通过 shrinker 回收
//    - 如 dentry cache、inode cache
```

## 二、LRU 链表

### 2.1 LRU 链表类型

```c
// include/linux/mmzone.h
enum lru_list {
    LRU_INACTIVE_ANON = 0,    // 不活跃匿名页
    LRU_ACTIVE_ANON   = 1,    // 活跃匿名页
    LRU_INACTIVE_FILE = 2,    // 不活跃文件页
    LRU_ACTIVE_FILE   = 3,    // 活跃文件页
    LRU_UNEVICTABLE   = 4,    // 不可驱逐页
    NR_LRU_LISTS
};

// 宏定义
#define for_each_lru(lru) for (lru = 0; lru < NR_LRU_LISTS; lru++)
#define for_each_evictable_lru(lru) for (lru = 0; lru <= LRU_ACTIVE_FILE; lru++)
```

### 2.2 LRU 链表结构

```
zone 或 memcg 的 lruvec：

┌─────────────────────────────────────────────────────────┐
│                      lruvec                              │
│                                                          │
│  ┌─────────────────┐  ┌─────────────────┐              │
│  │ INACTIVE_ANON   │  │  ACTIVE_ANON    │              │
│  │ ←────────────── │  │ ←────────────── │              │
│  │  冷 ← 页面 ← 热  │  │  冷 ← 页面 ← 热  │              │
│  └─────────────────┘  └─────────────────┘              │
│                                                          │
│  ┌─────────────────┐  ┌─────────────────┐              │
│  │ INACTIVE_FILE   │  │  ACTIVE_FILE    │              │
│  │ ←────────────── │  │ ←────────────── │              │
│  │  冷 ← 页面 ← 热  │  │  冷 ← 页面 ← 热  │              │
│  └─────────────────┘  └─────────────────┘              │
│                                                          │
│  ┌─────────────────┐                                    │
│  │   UNEVICTABLE   │  (mlock 锁定的页面)                │
│  └─────────────────┘                                    │
└─────────────────────────────────────────────────────────┘

页面生命周期：
1. 新页面加入 inactive 链表尾部（热端）
2. 被访问的页面设置 referenced 标志
3. 第二次访问时晋升到 active 链表
4. active 链表满时，老化页面降级到 inactive
5. inactive 链表的冷页面被回收
```

### 2.3 lruvec 数据结构

```c

    struct list_head lists[NR_LRU_LISTS];   // 5 个 LRU 链表
```

### 2.4 页面在 LRU 中的操作

```c
// mm/swap.c

// 添加页面到 LRU
void lru_cache_add(struct page *page)
{
    struct pagevec *pvec;
    
    // 使用 per-CPU pagevec 批量操作
    pvec = this_cpu_ptr(&lru_pvecs.lru_add);
    
    get_page(page);
    if (pagevec_add_and_need_flush(pvec, page))
        __pagevec_lru_add(pvec);
}

// 批量添加到 LRU
static void __pagevec_lru_add(struct pagevec *pvec)
{
    for (i = 0; i < pagevec_count(pvec); i++) {
        struct page *page = pvec->pages[i];
        struct lruvec *lruvec;
        
        lruvec = lock_page_lruvec_irqsave(page, &flags);
        __pagevec_lru_add_fn(page, lruvec);
        unlock_page_lruvec_irqrestore(lruvec, flags);
    }
    
    pagevec_reinit(pvec);
}

// 确定页面所属的 LRU 链表
static void __pagevec_lru_add_fn(struct page *page, struct lruvec *lruvec)
{
    enum lru_list lru;
    int was_unevictable = TestClearPageUnevictable(page);
    
    // 根据页面类型选择链表
    if (page_is_file_lru(page)) {
        // 文件页
        if (PageActive(page))
            lru = LRU_ACTIVE_FILE;
        else
            lru = LRU_INACTIVE_FILE;
    } else {
        // 匿名页
        if (PageActive(page))
            lru = LRU_ACTIVE_ANON;
        else
            lru = LRU_INACTIVE_ANON;
    }
    
    // 加入链表尾部
    list_add(&page->lru, &lruvec->lists[lru]);
    SetPageLRU(page);
}
```

### 2.5 页面激活

```c
// mm/swap.c
void mark_page_accessed(struct page *page)
{
    // 在 inactive 链表中被访问
    if (!PageActive(page) && !PageUnevictable(page) &&
        PageReferenced(page)) {
        // 晋升到 active 链表
        activate_page(page);
        ClearPageReferenced(page);
    } else if (!PageReferenced(page)) {
        // 第一次访问，设置 referenced
        SetPageReferenced(page);
    }
}

// 激活页面
static void activate_page(struct page *page)
{
    struct lruvec *lruvec;
    
    lruvec = lock_page_lruvec_irq(page);
    if (PageLRU(page) && !PageActive(page) && !PageUnevictable(page)) {
        // 从 inactive 移到 active
        del_page_from_lru_list(page, lruvec);
        SetPageActive(page);
        add_page_to_lru_list(page, lruvec);
    }
    unlock_page_lruvec_irq(lruvec);
}
```

## 三、kswapd 后台回收

### 3.1 kswapd 线程

```c
// mm/vmscan.c
// 每个 NUMA 节点一个 kswapd 线程

static int kswapd(void *p)
{
    pg_data_t *pgdat = (pg_data_t *)p;
    
    // 设置线程属性
    set_freezable();
    
    while (!kthread_should_stop()) {
        bool ret;
        
        // 等待唤醒
        kswapd_try_to_sleep(pgdat, highest_zoneidx);
        
        // 检查是否需要回收
        if (!pgdat_balanced(pgdat, order, highest_zoneidx)) {
            // 执行回收
            balance_pgdat(pgdat, order, highest_zoneidx);
        }
    }
    
    return 0;
}
```

### 3.2 kswapd 唤醒条件

```c
// mm/page_alloc.c
// 内存分配时检查是否需要唤醒 kswapd

static inline bool zone_watermark_ok(struct zone *z, unsigned int order,
                                     unsigned long mark, int highest_zoneidx,
                                     unsigned int alloc_flags)
{
    return __zone_watermark_ok(z, order, mark, highest_zoneidx, alloc_flags,
                               zone_page_state(z, NR_FREE_PAGES));
}

// 唤醒 kswapd
void wakeup_kswapd(struct zone *zone, gfp_t gfp_flags, int order,
                   enum zone_type highest_zoneidx)
{
    pg_data_t *pgdat;
    
    if (!managed_zone(zone))
        return;
    
    // 检查是否低于 low_wmark
    if (!pgdat_balanced(pgdat, order, highest_zoneidx)) {
        pgdat->kswapd_highest_zoneidx = highest_zoneidx;
        pgdat->kswapd_order = order;
        wake_up_interruptible(&pgdat->kswapd_wait);
    }
}
```

### 3.3 balance_pgdat - 平衡节点内存

```c
// mm/vmscan.c
static int balance_pgdat(pg_data_t *pgdat, int order, int highest_zoneidx)
{
    struct scan_control sc = {
        .gfp_mask = GFP_KERNEL,
        .order = order,
        .may_unmap = 1,
        .may_swap = 1,
        .may_writepage = !laptop_mode,
    };
    
    // 优先级从高到低
    for (sc.priority = DEF_PRIORITY; sc.priority >= 0; sc.priority--) {
        bool balanced;
        
        // 收缩节点
        shrink_node(pgdat, &sc);
        
        // 检查是否达到目标
        balanced = pgdat_balanced(pgdat, order, highest_zoneidx);
        if (balanced)
            break;
        
        // 如果还不够，降低优先级继续
    }
    
    return sc.order;
}
```

## 四、页面回收核心流程

### 4.1 shrink_node

```c
// mm/vmscan.c
static void shrink_node(pg_data_t *pgdat, struct scan_control *sc)
{
    struct lruvec *target_lruvec;
    unsigned long nr_reclaimed, nr_scanned;
    
    target_lruvec = mem_cgroup_lruvec(sc->target_mem_cgroup, pgdat);
    
    do {
        // 决定扫描匿名页和文件页的比例
        get_scan_count(target_lruvec, sc, nr);
        
        // 收缩 LRU
        shrink_lruvec(target_lruvec, sc);
        
        // 收缩 slab
        shrink_slab(sc->gfp_mask, pgdat->node_id, NULL, sc->priority);
        
    } while (should_continue_reclaim(pgdat, sc));
}
```

### 4.2 get_scan_count - 决定扫描比例

```c
// mm/vmscan.c
static void get_scan_count(struct lruvec *lruvec, struct scan_control *sc,
                           unsigned long *nr)
{
    unsigned long anon_cost, file_cost, total_cost;
    unsigned long ap, fp;
    
    // 根据 swappiness 和访问频率计算比例
    // swappiness 越高，越倾向于回收匿名页
    
    anon_cost = lruvec->anon_cost;
    file_cost = lruvec->file_cost;
    
    // 默认 swappiness = 60
    // 匿名页扫描 = 总扫描 * swappiness / 100
    // 文件页扫描 = 总扫描 * (100 - swappiness) / 100
    
    // 加入访问代价因素
    ap = swappiness * (total_cost + 1) / (anon_cost + 1);
    fp = (200 - swappiness) * (total_cost + 1) / (file_cost + 1);
    
    // 分配扫描数量
    nr[LRU_INACTIVE_ANON] = scan * ap / (ap + fp);
    nr[LRU_ACTIVE_ANON] = scan * ap / (ap + fp);
    nr[LRU_INACTIVE_FILE] = scan * fp / (ap + fp);
    nr[LRU_ACTIVE_FILE] = scan * fp / (ap + fp);
}
```

### 4.3 shrink_lruvec - 收缩 LRU

```c
// mm/vmscan.c
static void shrink_lruvec(struct lruvec *lruvec, struct scan_control *sc)
{
    unsigned long nr_reclaimed = 0;
    enum lru_list lru;
    
    while (nr[LRU_INACTIVE_ANON] || nr[LRU_ACTIVE_ANON] ||
           nr[LRU_INACTIVE_FILE] || nr[LRU_ACTIVE_FILE]) {
        
        // 先收缩不活跃链表
        for_each_evictable_lru(lru) {
            if (nr[lru]) {
                nr_to_scan = min(nr[lru], SWAP_CLUSTER_MAX);
                nr[lru] -= nr_to_scan;
                
                nr_reclaimed += shrink_list(lru, nr_to_scan, lruvec, sc);
            }
        }
        
        if (sc->nr_reclaimed >= sc->nr_to_reclaim)
            break;
    }
}
```

### 4.4 shrink_inactive_list - 回收不活跃页面

```c
// mm/vmscan.c
static unsigned long shrink_inactive_list(unsigned long nr_to_scan,
                                          struct lruvec *lruvec,
                                          struct scan_control *sc,
                                          enum lru_list lru)
{
    LIST_HEAD(page_list);
    unsigned long nr_reclaimed = 0;
    unsigned long nr_taken;
    
    // 1. 从 LRU 尾部隔离页面
    spin_lock_irq(&lruvec->lru_lock);
    nr_taken = isolate_lru_pages(nr_to_scan, lruvec, &page_list, ...);
    spin_unlock_irq(&lruvec->lru_lock);
    
    if (nr_taken == 0)
        return 0;
    
    // 2. 回收页面
    nr_reclaimed = shrink_page_list(&page_list, pgdat, sc, ...);
    
    // 3. 未回收的页面放回 LRU
    spin_lock_irq(&lruvec->lru_lock);
    putback_inactive_pages(lruvec, &page_list);
    spin_unlock_irq(&lruvec->lru_lock);
    
    return nr_reclaimed;
}
```

### 4.5 shrink_page_list - 回收页面列表

```c
// mm/vmscan.c
static unsigned long shrink_page_list(struct list_head *page_list,
                                      struct pglist_data *pgdat,
                                      struct scan_control *sc, ...)
{
    unsigned long nr_reclaimed = 0;
    
    while (!list_empty(page_list)) {
        struct page *page = lru_to_page(page_list);
        
        // 锁定页面
        if (!trylock_page(page))
            goto keep;
        
        // 1. 检查是否被引用
        if (page_referenced(page, 0, sc->target_mem_cgroup, &vm_flags)) {
            // 最近被访问，放回 active 链表
            goto activate_locked;
        }
        
        // 2. 匿名页需要换出
        if (PageAnon(page) && !PageSwapCache(page)) {
            if (!sc->may_swap)
                goto keep_locked;
            
            // 添加到 swap cache
            if (!add_to_swap(page))
                goto activate_locked;
        }
        
        // 3. 解除所有映射
        if (page_mapped(page)) {
            if (!try_to_unmap(page, TTU_BATCH_FLUSH))
                goto keep_locked;
        }
        
        // 4. 脏页需要回写
        if (PageDirty(page)) {
            // 触发回写
            pageout(page, mapping);
            goto keep;
        }
        
        // 5. 从 Page Cache 移除
        if (page_has_private(page)) {
            if (!try_to_release_page(page, sc->gfp_mask))
                goto keep_locked;
        }
        
        // 6. 释放页面
        __delete_from_page_cache(page, NULL);
        __ClearPageLocked(page);
        
        free_unref_page(page);
        nr_reclaimed++;
        continue;
        
activate_locked:
        SetPageActive(page);
keep_locked:
        unlock_page(page);
keep:
        list_add(&page->lru, &ret_pages);
    }
    
    return nr_reclaimed;
}
```

## 五、内存水位线

### 5.1 水位线计算

```c
// mm/page_alloc.c
void __setup_per_zone_wmarks(void)
{
    unsigned long pages_min = min_free_kbytes >> (PAGE_SHIFT - 10);
    unsigned long pages_low, pages_high;
    
    for_each_zone(zone) {
        unsigned long min;
        
        // 按比例计算
        min = (pages_min * zone->managed_pages) / lowmem_pages;
        
        // 设置水位线
        zone->_watermark[WMARK_MIN] = min;
        zone->_watermark[WMARK_LOW] = min + min / 4;   // min * 1.25
        zone->_watermark[WMARK_HIGH] = min + min / 2;  // min * 1.5
    }
}

// min_free_kbytes 计算（启动时）
// 大约为物理内存的 0.5%，但有上下限
// 默认范围：16MB - 64MB
```

### 5.2 水位线检查

```c
// include/linux/mmzone.h
static inline bool zone_watermark_ok(struct zone *z, unsigned int order,
                                     unsigned long mark, int highest_zoneidx,
                                     unsigned int alloc_flags)
{
    return __zone_watermark_ok(z, order, mark, highest_zoneidx, alloc_flags,
                               zone_page_state(z, NR_FREE_PAGES));
}

// mm/page_alloc.c
bool __zone_watermark_ok(struct zone *z, unsigned int order, unsigned long mark,
                         int highest_zoneidx, unsigned int alloc_flags,
                         long free_pages)
{
    long min = mark;
    
    // 保留给高优先级分配
    if (alloc_flags & ALLOC_HIGH)
        min -= min / 2;
    if (alloc_flags & ALLOC_HARDER)
        min -= min / 4;
    
    // 检查空闲页数是否足够
    if (free_pages <= min + z->lowmem_reserve[highest_zoneidx])
        return false;
    
    // 检查是否有足够大小的连续块
    for (o = 0; o < order; o++) {
        free_pages -= z->free_area[o].nr_free << o;
        min >>= 1;
        if (free_pages <= min)
            return false;
    }
    
    return true;
}
```

## 总结

### 核心概念

1. **LRU 链表**：按访问热度管理页面
2. **kswapd**：后台内存回收线程
3. **水位线**：控制回收触发的阈值
4. **swappiness**：控制匿名页和文件页的回收比例

### 回收流程

```
内存压力 → 唤醒 kswapd → balance_pgdat → shrink_node
                                              │
                    ┌─────────────────────────┼─────────────────────────┐
                    │                         │                         │
                    ▼                         ▼                         ▼
             shrink_lruvec             shrink_slab                 ...
                    │
                    ▼
           shrink_inactive_list
                    │
                    ▼
            shrink_page_list
                    │
         ┌──────────┼──────────┐
         │          │          │
         ▼          ▼          ▼
    解除映射    回写脏页    释放页面
```
