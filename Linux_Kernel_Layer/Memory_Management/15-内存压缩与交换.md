## 学习目标

- 理解 swap 机制的原理和作用

- 了解 Android 中的内存压缩策略

## 一、Swap 概述

### 1.1 什么是 Swap

Swap 是将匿名页面换出到外部存储（磁盘或其他介质）的机制：

```
                    物理内存                        Swap 空间
┌────────────────────────────────┐      ┌────────────────────────┐
│                                │      │                        │
│  ┌────┐ ┌────┐ ┌────┐        │      │ ┌────┐ ┌────┐ ┌────┐  │
│  │活跃│ │活跃│ │活跃│        │      │ │页A │ │页B │ │页C │  │
│  │页面│ │页面│ │页面│        │      │ │    │ │    │ │    │  │
│  └────┘ └────┘ └────┘        │      │ └────┘ └────┘ └────┘  │
│                                │      │                        │
│  ┌────┐ ┌────┐                │ 换出  │                        │
│  │冷页│ │冷页│ ────────────────┼─────►│                        │
│  │面A │ │面B │                │      │                        │
│  └────┘ └────┘                │      │                        │
│                                │ 换入  │                        │
│         ◄──────────────────────┼─────│                        │
│                                │      │                        │
└────────────────────────────────┘      └────────────────────────┘
```

### 1.2 Swap 类型

| 类型 | 介质 | 速度 | 用途 |
|-----|------|------|-----|
| Swap 分区 | 磁盘分区 | 较快 | 传统服务器 |
| Swap 文件 | 文件系统文件 | 较慢 | 灵活配置 |
| zRAM | 压缩内存 | 最快 | Android/嵌入式 |
| zswap | 压缩缓存 | 快 | 压缩 + 磁盘 |

### 1.3 Android 中的 Swap

Android 主要使用 zRAM：
- 不使用磁盘 swap（闪存寿命考虑）
- 压缩匿名页面存储在内存中
- 压缩率约 2:1 到 3:1

### 2.2 Swap Entry

```c
// include/linux/swapops.h
// Swap entry 格式（存储在 PTE 中）
/*
 * 64位系统：
 * ┌─────────────────────────────────────────────────────────┐
 * │ offset (50 bits) │ type (5 bits) │ present (1 bit) = 0 │
 * └─────────────────────────────────────────────────────────┘
 *
 * offset: 在 swap 空间中的位置
 * type: swap 设备编号
 * present: 0 表示不在内存中
 */

typedef struct {
    unsigned long val;
} swp_entry_t;

// 操作宏
#define swp_type(entry)     (((entry).val >> SWP_TYPE_SHIFT) & SWP_TYPE_MASK)
#define swp_offset(entry)   ((entry).val >> SWP_OFFSET_SHIFT)
#define swp_entry(type, offset) \
    ((swp_entry_t) { ((type) << SWP_TYPE_SHIFT) | ((offset) << SWP_OFFSET_SHIFT) })
```

### 2.3 换出流程

```c
// mm/vmscan.c
// 在 shrink_page_list 中触发换出

// 1. 检查是否是匿名页
if (PageAnon(page) && !PageSwapCache(page)) {
    // 2. 分配 swap slot
    if (!add_to_swap(page))
        goto activate_locked;
}

// mm/swap_state.c
int add_to_swap(struct page *page)
{
    swp_entry_t entry;
    
    // 分配 swap entry
    entry = get_swap_page(page);
    if (!entry.val)
        return 0;
    
    // 添加到 swap cache
    if (add_to_swap_cache(page, entry, __GFP_HIGH | __GFP_NOMEMALLOC) == 0) {
        // 设置页面脏标志，触发回写
        SetPageDirty(page);
        return 1;
    }
    
    put_swap_page(page, entry);
    return 0;
}

// 实际写入 swap
// mm/page_io.c
int swap_writepage(struct page *page, struct writeback_control *wbc)
{
    // 构建 bio 并提交
    struct bio *bio = get_swap_bio(GFP_NOIO, page, end_swap_bio_write);
    
    bio_set_op_attrs(bio, REQ_OP_WRITE, REQ_SWAP);
    
    submit_bio(bio);
    
    return 0;
}
```

### 2.4 换入流程

```c
// mm/memory.c
// 在缺页处理中触发换入

vm_fault_t do_swap_page(struct vm_fault *vmf)
{
    swp_entry_t entry;
    struct page *page;
    
    // 获取 swap entry
    entry = pte_to_swp_entry(vmf->orig_pte);
    
    // 1. 检查 swap cache
    page = lookup_swap_cache(entry, vma, vmf->address);
    if (page)
        goto have_page;
    
    // 2. 不在 cache 中，从 swap 读取
    page = swapin_readahead(entry, GFP_HIGHUSER_MOVABLE, vmf);
    if (!page) {
        // 同步读取单页
        page = read_swap_cache_async(entry, GFP_HIGHUSER_MOVABLE,
                                     vma, vmf->address, false);
    }
    
have_page:
    // 3. 建立页表映射
    set_pte_at(vma->vm_mm, vmf->address, vmf->pte, pte);
    
    // 4. 释放 swap entry
    swap_free(entry);
    
    return 0;
}

// mm/page_io.c
int swap_readpage(struct page *page, bool synchronous)
{
    struct bio *bio;
    
    // 构建 bio
    bio = get_swap_bio(GFP_KERNEL, page, end_swap_bio_read);
    bio_set_op_attrs(bio, REQ_OP_READ, 0);
    
    // 提交 I/O
    submit_bio(bio);
    
    if (synchronous) {
        // 同步等待
        wait_on_page_locked(page);
    }
    
    return 0;
}
```

## 三、zRAM 内存压缩

### 3.1 zRAM 原理

```
                    压缩前                              压缩后
┌────────────────────────────────┐      ┌────────────────────────────┐
│         匿名页面               │      │        zRAM 设备            │
│                                │      │                            │
│  ┌────┐ ┌────┐ ┌────┐ ┌────┐ │      │  ┌──┐ ┌──┐ ┌──┐ ┌──┐      │
│  │4KB │ │4KB │ │4KB │ │4KB │ │      │  │2KB│ │1KB│ │2KB│ │1KB│     │
│  │    │ │    │ │    │ │    │ │ ───► │  │   │ │   │ │   │ │   │     │
│  └────┘ └────┘ └────┘ └────┘ │      │  └──┘ └──┘ └──┘ └──┘      │
│                                │      │                            │
│        16KB 物理内存           │      │     6KB 压缩后             │
└────────────────────────────────┘      └────────────────────────────┘

压缩率：约 2:1 到 3:1（取决于数据内容）
```

### 3.2 zRAM 配置


### 3.3 zRAM 内核实现

```c
// drivers/block/zram/zram_drv.c

struct zram {
    struct zram_table_entry *table;  // 页面表
    struct zs_pool *mem_pool;        // zsmalloc 内存池
    struct zcomp *comp;              // 压缩器
    struct gendisk *disk;            // 块设备
    struct rw_semaphore init_lock;
    
    // 统计
    atomic64_t stats[NR_ZRAM_STAT_ITEM];
    
    // 配置
    u64 disksize;                    // 设备大小
    char compressor[CRYPTO_MAX_ALG_NAME]; // 压缩算法
};

// 写入（压缩）
static int zram_bvec_write(struct zram *zram, struct bio_vec *bvec,
                           u32 index, int offset, struct bio *bio)
{
    struct zcomp_strm *zstrm;
    unsigned long handle;
    unsigned int comp_len;
    void *src, *dst;
    
    // 获取压缩流
    zstrm = zcomp_stream_get(zram->comp);
    
    // 压缩页面
    src = kmap_atomic(page);
    ret = zcomp_compress(zstrm, src, &comp_len);
    kunmap_atomic(src);
    
    // 分配压缩后的存储空间
    handle = zs_malloc(zram->mem_pool, comp_len, GFP_NOIO | __GFP_NOWARN);
    
    // 存储压缩数据
    dst = zs_map_object(zram->mem_pool, handle, ZS_MM_WO);
    memcpy(dst, zstrm->buffer, comp_len);
    zs_unmap_object(zram->mem_pool, handle);
    
    // 记录映射
    zram->table[index].handle = handle;
    zram->table[index].flags = comp_len;
    
    return 0;
}

// 读取（解压）
static int zram_bvec_read(struct zram *zram, struct bio_vec *bvec,
                          u32 index, int offset, struct bio *bio)
{
    void *src, *dst;
    unsigned long handle;
    unsigned int size;
    
    // 获取压缩数据
    handle = zram->table[index].handle;
    size = zram->table[index].flags;
    
    src = zs_map_object(zram->mem_pool, handle, ZS_MM_RO);
    
    // 解压
    dst = kmap_atomic(page);
    ret = zcomp_decompress(zstrm, src, size, dst);
    kunmap_atomic(dst);
    
    zs_unmap_object(zram->mem_pool, handle);
    
    return 0;
}
```

### 3.4 压缩算法

```c
// 支持的压缩算法
// LZ4:    速度最快，压缩率中等
// LZO:    速度快，压缩率中等
// ZSTD:   速度较快，压缩率高
// Deflate: 速度慢，压缩率高

// Android 常用 LZ4
// /sys/block/zram0/comp_algorithm 显示可用算法

// 压缩接口
// drivers/block/zram/zcomp.c
int zcomp_compress(struct zcomp_strm *zstrm,
                   const void *src, unsigned int *dst_len)
{
    // 调用底层压缩算法
    return crypto_comp_compress(zstrm->tfm, src, PAGE_SIZE,
                                zstrm->buffer, dst_len);
}

int zcomp_decompress(struct zcomp_strm *zstrm,
                     const void *src, unsigned int src_len, void *dst)
{
    unsigned int dst_len = PAGE_SIZE;
    return crypto_comp_decompress(zstrm->tfm, src, src_len,
                                  dst, &dst_len);
}
```

### 4.3 LMKD 使用 PSI

```c
// Android LMKD 使用 PSI 监控内存压力
// system/memory/lmkd/lmkd.cpp

// 打开 PSI 监控
int init_psi_monitor(enum psi_stall_type stall_type,
                     int threshold_us, int window_us)
{
    int fd = open("/proc/pressure/memory", O_RDWR | O_CLOEXEC);
    
    // 配置触发条件
    char buf[64];
    snprintf(buf, sizeof(buf), "%s %d %d",
             stall_type == PSI_SOME ? "some" : "full",
             threshold_us, window_us);
    
    write(fd, buf, strlen(buf));
    
    return fd;
}

// 监控循环
void monitor_psi() {
    struct pollfd fds[1];
    fds[0].fd = psi_fd;
    fds[0].events = POLLPRI;
    
    while (true) {
        int ret = poll(fds, 1, -1);
        if (ret > 0 && (fds[0].revents & POLLPRI)) {
            // 内存压力触发，执行杀进程
            handle_memory_pressure();
        }
    }
}
```

## 五、zswap

### 5.1 zswap 原理

zswap 是在 swap 和内存之间的压缩缓存层：

```
               ┌─────────────┐
               │   匿名页    │
               └──────┬──────┘
                      │ 换出
                      ▼
               ┌─────────────┐
               │   zswap     │ ← 压缩缓存
               │  (内存)     │
               └──────┬──────┘
                      │ 缓存满时写入
                      ▼
               ┌─────────────┐
               │ 磁盘 swap   │
               └─────────────┘

优点：
- 减少磁盘 I/O
- 利用压缩减少实际写入量
- 换入时从内存读取更快
```

### 5.2 zswap vs zRAM

| 特性 | zRAM | zswap |
|-----|------|-------|
| 后备存储 | 无 | 磁盘 swap |
| 容量 | 固定 | 弹性（可溢出到磁盘）|
| 适用场景 | 无磁盘系统 | 有磁盘系统 |
| Android | 常用 | 不常用 |

## 总结

### 核心概念

1. **Swap**：将匿名页换出到外部存储
2. **zRAM**：压缩内存作为 swap 设备
3. **zswap**：swap 的压缩缓存层
4. **PSI**：内存压力监控机制
5. **swappiness**：控制换出倾向

### Android 内存压缩特点

- 使用 zRAM 而非磁盘 swap
- LMKD 使用 PSI 监控内存压力
- 压缩算法通常使用 LZ4
