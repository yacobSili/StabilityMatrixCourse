## 一、mmap 概述

### 1.1 什么是内存映射

内存映射（Memory Mapping）是将文件或设备映射到进程地址空间的技术：

```
传统文件 I/O：
┌─────────┐    read()    ┌─────────┐    copy    ┌─────────┐
│  磁盘   │ ───────────► │Page Cache│ ───────► │用户缓冲区│
└─────────┘              └─────────┘           └─────────┘

内存映射：
┌─────────┐    read()    ┌─────────┐
│  磁盘   │ ───────────► │Page Cache│ ◄──── 直接访问
└─────────┘              └─────────┘           │
                              ▲                │
                              │ mmap           │
                              │                │
                         ┌────┴────┐           │
                         │  VMA    │───────────┘
                         └─────────┘
                         进程地址空间
```

### 1.2 mmap 的优势

| 优势 | 说明 |
|-----|------|
| 减少数据拷贝 | 直接访问 Page Cache，无需用户态缓冲 |
| 简化编程 | 像访问内存一样访问文件 |
| 进程间共享 | 多进程可共享同一映射 |
| 按需加载 | 缺页时才加载，减少初始开销 |
| 利用虚拟内存 | 可映射大于物理内存的文件 |

### 1.3 映射类型

```
                    ┌─────────────────┐
                    │     mmap()      │
                    └────────┬────────┘
                             │
              ┌──────────────┴──────────────┐
              │                              │
       ┌──────┴──────┐                ┌──────┴──────┐
       │  文件映射    │                │  匿名映射    │
       │ (File-backed)│                │(Anonymous)  │
       └──────┬──────┘                └──────┬──────┘
              │                              │
       ┌──────┴──────┐                ┌──────┴──────┐
       │             │                │             │
   ┌───┴───┐    ┌───┴───┐        ┌───┴───┐    ┌───┴───┐
   │ 共享  │    │ 私有  │        │ 共享  │    │ 私有  │
   │MAP_   │    │MAP_   │        │MAP_   │    │MAP_   │
   │SHARED │    │PRIVATE│        │SHARED │    │PRIVATE│
   └───────┘    └───────┘        └───────┘    └───────┘
```

## 二、mmap 系统调用

### 2.1 函数原型

```c
#include <sys/mman.h>

void *mmap(void *addr, size_t length, int prot, int flags,
           int fd, off_t offset);

int munmap(void *addr, size_t length);

// 参数说明：
// addr    - 建议的映射地址（通常为 NULL）
// length  - 映射长度
// prot    - 保护属性（PROT_READ/WRITE/EXEC/NONE）
// flags   - 映射标志
// fd      - 文件描述符（匿名映射为 -1）
// offset  - 文件偏移（必须页对齐）
```

### 2.2 保护属性 (prot)

```c
#define PROT_READ       0x1     // 可读
#define PROT_WRITE      0x2     // 可写
#define PROT_EXEC       0x4     // 可执行
#define PROT_NONE       0x0     // 不可访问

// 常用组合
PROT_READ                       // 只读
PROT_READ | PROT_WRITE          // 读写
PROT_READ | PROT_EXEC           // 可读可执行
```

### 2.3 映射标志 (flags)

```c
// 必选（二选一）
#define MAP_SHARED      0x01    // 共享映射
#define MAP_PRIVATE     0x02    // 私有映射（COW）

// 可选标志
#define MAP_FIXED       0x10    // 强制使用指定地址
#define MAP_ANONYMOUS   0x20    // 匿名映射
#define MAP_GROWSDOWN   0x0100  // 向下增长（栈）
#define MAP_LOCKED      0x2000  // 锁定在内存中
#define MAP_NORESERVE   0x4000  // 不预留 swap 空间
#define MAP_POPULATE    0x8000  // 预先填充页表
#define MAP_NONBLOCK    0x10000 // 非阻塞（与 POPULATE 配合）
#define MAP_HUGETLB     0x40000 // 使用大页

// Linux 特有
#define MAP_STACK       0x20000 // 用于栈
#define MAP_SYNC        0x80000 // DAX 同步映射
```

### 2.4 使用示例

```c
// 1. 文件映射（只读）
int fd = open("file.txt", O_RDONLY);
struct stat st;
fstat(fd, &st);
char *data = mmap(NULL, st.st_size, PROT_READ, MAP_PRIVATE, fd, 0);
// 访问 data[0..st.st_size-1]
munmap(data, st.st_size);
close(fd);

// 2. 文件映射（读写，共享）
int fd = open("file.txt", O_RDWR);
char *data = mmap(NULL, 4096, PROT_READ | PROT_WRITE, MAP_SHARED, fd, 0);
data[0] = 'X';  // 修改会写入文件
msync(data, 4096, MS_SYNC);  // 同步到磁盘
munmap(data, 4096);
close(fd);

// 3. 匿名映射
void *mem = mmap(NULL, 4096, PROT_READ | PROT_WRITE,
                 MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
// 使用内存
munmap(mem, 4096);

// 4. 进程间共享内存
int fd = shm_open("/my_shm", O_CREAT | O_RDWR, 0600);
ftruncate(fd, 4096);
void *shared = mmap(NULL, 4096, PROT_READ | PROT_WRITE, MAP_SHARED, fd, 0);
// 多进程可共享 shared
munmap(shared, 4096);
```

## 三、文件映射

### 3.1 文件映射原理

```
文件映射内部结构：

进程地址空间                    内核
┌──────────────┐
│     VMA      │
│  vm_file ────┼──────────► struct file
│  vm_pgoff    │                │
└──────────────┘                ▼
                          struct inode
                                │
                                ▼
                         address_space
                                │
                                ▼
┌───────────────────────────────┴───────────────────────────────┐
│                         Page Cache                             │
│  ┌─────┐ ┌─────┐ ┌─────┐ ┌─────┐ ┌─────┐                      │
│  │page0│ │page1│ │page2│ │page3│ │ ... │                      │
│  └─────┘ └─────┘ └─────┘ └─────┘ └─────┘                      │
└───────────────────────────────────────────────────────────────┘
```

### 3.2 私有文件映射 (MAP_PRIVATE)

```c
// 特点：
// - 写时复制（COW）
// - 修改不影响文件
// - 其他进程看不到修改

// 使用场景：
// - 加载可执行文件代码段
// - 加载共享库

// 示例：加载可执行文件
int fd = open("/bin/ls", O_RDONLY);
// 代码段：只读私有映射
void *text = mmap(NULL, text_size, PROT_READ | PROT_EXEC,
                  MAP_PRIVATE, fd, text_offset);
// 数据段：读写私有映射（COW）
void *data = mmap(NULL, data_size, PROT_READ | PROT_WRITE,
                  MAP_PRIVATE, fd, data_offset);
```

**私有映射写入流程**：

```
初始状态：
┌──────────────┐           ┌──────────────┐
│   进程 A     │           │   进程 B     │
│   PTE: R     │──────┬───►│   PTE: R     │
└──────────────┘      │    └──────────────┘
                      │
                      ▼
                ┌──────────┐
                │Page Cache│
                └──────────┘

进程 A 写入后（COW）：
┌──────────────┐           ┌──────────────┐
│   进程 A     │           │   进程 B     │
│   PTE: RW    │           │   PTE: R     │
└──────┬───────┘           └──────┬───────┘
       │                          │
       ▼                          ▼
┌──────────────┐           ┌──────────────┐
│ 私有副本      │           │Page Cache   │
│ (匿名页)      │           │ (不变)       │
└──────────────┘           └──────────────┘
```

### 3.3 共享文件映射 (MAP_SHARED)

```c
// 特点：
// - 多进程共享同一物理页
// - 修改对所有进程可见
// - 修改会写回文件

// 使用场景：
// - 进程间通信
// - 数据库文件访问
// - 日志文件

// 示例：进程间共享
// 进程 A
int fd = open("shared.dat", O_RDWR);
int *shared = mmap(NULL, 4096, PROT_READ | PROT_WRITE,
                   MAP_SHARED, fd, 0);
shared[0] = 42;  // 其他进程可见

// 进程 B
int fd = open("shared.dat", O_RDWR);
int *shared = mmap(NULL, 4096, PROT_READ | PROT_WRITE,
                   MAP_SHARED, fd, 0);
printf("%d\n", shared[0]);  // 输出 42
```

**共享映射同步**：

```c
// msync - 同步映射区域到文件
int msync(void *addr, size_t length, int flags);

// flags:
// MS_SYNC    - 同步写入，等待完成
// MS_ASYNC   - 异步写入，立即返回
// MS_INVALIDATE - 使其他映射失效

msync(shared, 4096, MS_SYNC);  // 确保写入磁盘
```

## 四、匿名映射

### 4.1 匿名映射原理

```c
// 匿名映射：不关联任何文件
// 初始内容为零
// 常用于大内存分配

void *mem = mmap(NULL, size,
                 PROT_READ | PROT_WRITE,
                 MAP_PRIVATE | MAP_ANONYMOUS,
                 -1, 0);

// 等价于（Linux 特有）
int fd = open("/dev/zero", O_RDWR);
void *mem = mmap(NULL, size, PROT_READ | PROT_WRITE,
                 MAP_PRIVATE, fd, 0);
close(fd);
```

### 4.2 私有匿名映射

```c
// 特点：
// - 进程私有
// - 初始为零
// - fork 后 COW

// 使用场景：
// - malloc 大内存分配
// - 线程栈

// glibc malloc 使用 mmap 分配大块内存
// 阈值通常为 128KB（可通过 M_MMAP_THRESHOLD 配置）

// 内核实现
static vm_fault_t do_anonymous_page(struct vm_fault *vmf)
{
    // 首次读：映射零页（共享）
    // 首次写：分配新页面，清零
}
```

### 4.3 共享匿名映射

```c
// 特点：
// - fork 后父子进程共享
// - 用于进程间通信

// 使用场景：
// - 父子进程共享内存

void *shared = mmap(NULL, 4096,
                    PROT_READ | PROT_WRITE,
                    MAP_SHARED | MAP_ANONYMOUS,
                    -1, 0);

pid_t pid = fork();
if (pid == 0) {
    // 子进程
    *(int *)shared = 42;  // 父进程可见
} else {
    // 父进程
    wait(NULL);
    printf("%d\n", *(int *)shared);  // 输出 42
}
```

## 五、内核实现

### 5.1 mmap 系统调用实现

```c
// mm/mmap.c
SYSCALL_DEFINE6(mmap, unsigned long, addr, unsigned long, len,
                unsigned long, prot, unsigned long, flags,
                unsigned long, fd, unsigned long, off)
{
    return ksys_mmap_pgoff(addr, len, prot, flags, fd, off >> PAGE_SHIFT);
}

unsigned long ksys_mmap_pgoff(unsigned long addr, unsigned long len,
                              unsigned long prot, unsigned long flags,
                              unsigned long fd, unsigned long pgoff)
{
    struct file *file = NULL;
    
    // 匿名映射
    if (!(flags & MAP_ANONYMOUS)) {
        // 获取文件引用
        file = fget(fd);
        if (!file)
            return -EBADF;
    }
    
    // 核心映射函数
    retval = vm_mmap_pgoff(file, addr, len, prot, flags, pgoff);
    
    if (file)
        fput(file);
    
    return retval;
}
```

### 5.2 do_mmap 核心实现

```c
unsigned long do_mmap(struct file *file, unsigned long addr,
                      unsigned long len, unsigned long prot,
                      unsigned long flags, unsigned long pgoff,
                      unsigned long *populate, struct list_head *uf)
{
    struct mm_struct *mm = current->mm;
    vm_flags_t vm_flags;
    
    // 1. 计算 VM 标志
    vm_flags = calc_vm_prot_bits(prot) | calc_vm_flag_bits(flags);
    
    // 2. 文件映射额外处理
    if (file) {
        // 检查文件权限
        if (!file->f_op->mmap)
            return -ENODEV;
        
        // 共享写映射需要写权限
        if ((vm_flags & (VM_SHARED | VM_WRITE)) == (VM_SHARED | VM_WRITE)) {
            if (!(file->f_mode & FMODE_WRITE))
                return -EACCES;
        }
    }
    
    // 3. 查找合适的地址
    addr = get_unmapped_area(file, addr, len, pgoff, flags);
    if (IS_ERR_VALUE(addr))
        return addr;
    
    // 4. 创建 VMA
    addr = mmap_region(file, addr, len, vm_flags, pgoff, uf);
    
    // 5. MAP_POPULATE 预填充
    if (populate)
        *populate = (vm_flags & VM_LOCKED) ? len : 0;
    
    return addr;
}
```

### 5.3 文件映射的 vm_ops

```c
// 文件系统提供 vm_ops
// fs/ext4/file.c
const struct vm_operations_struct ext4_file_vm_ops = {
    .fault          = ext4_filemap_fault,   // 缺页处理
    .map_pages      = filemap_map_pages,    // 批量映射
    .page_mkwrite   = ext4_page_mkwrite,    // 写时处理
};

// 通用文件映射缺页处理
// mm/filemap.c
vm_fault_t filemap_fault(struct vm_fault *vmf)
{
    struct file *file = vmf->vma->vm_file;
    struct address_space *mapping = file->f_mapping;
    pgoff_t offset = vmf->pgoff;
    struct page *page;
    
    // 从 Page Cache 获取页面
    page = find_get_page(mapping, offset);
    if (!page) {
        // 不在缓存中，需要读取
        page = pagecache_get_page(mapping, offset, FGP_CREAT, vmf->gfp_mask);
        if (!PageUptodate(page)) {
            // 从文件读取
            mapping->a_ops->readpage(file, page);
        }
    }
    
    vmf->page = page;
    return VM_FAULT_LOCKED;
}
```

## 六、Android 中的内存映射应用

### 6.1 Binder 使用 mmap

```c
// Binder 驱动使用 mmap 实现高效 IPC
// drivers/android/binder.c

static int binder_mmap(struct file *filp, struct vm_area_struct *vma)
{
    struct binder_proc *proc = filp->private_data;
    
    // Binder 只允许只读映射
    if ((vma->vm_flags & VM_WRITE) && (vma->vm_flags & VM_READ))
        return -EPERM;
    
    // 分配 Binder 缓冲区
    proc->buffer = kzalloc(sizeof(*proc->buffer), GFP_KERNEL);
    
    // 创建用户空间映射
    // 数据只需要一次拷贝（内核到用户空间）
    vma->vm_ops = &binder_vm_ops;
    
    return 0;
}
```

## 总结

### 映射类型对比

| 类型 | 标志 | 特点 | 用途 |
|-----|------|------|-----|
| 私有文件映射 | MAP_PRIVATE | COW，不影响文件 | 加载可执行文件 |
| 共享文件映射 | MAP_SHARED | 修改写回文件 | IPC，数据库 |
| 私有匿名映射 | MAP_PRIVATE\|MAP_ANONYMOUS | 进程私有 | malloc 大块分配 |
| 共享匿名映射 | MAP_SHARED\|MAP_ANONYMOUS | fork 后共享 | 父子进程 IPC |

### 关键函数

| 函数 | 作用 |
|-----|------|
| mmap() | 创建内存映射 |
| munmap() | 解除映射 |
| msync() | 同步到文件 |
| mprotect() | 修改保护属性 |
| mremap() | 调整映射大小 |
