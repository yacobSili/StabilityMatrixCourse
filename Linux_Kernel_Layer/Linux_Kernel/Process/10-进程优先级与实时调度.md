# 进程优先级与实时调度

## 学习目标

- 理解 Linux 进程优先级体系
- 掌握 nice 值与静态优先级的关系
- 理解实时调度策略（SCHED_FIFO、SCHED_RR）
- 了解 Deadline 调度（SCHED_DEADLINE）
- 理解 Android 进程优先级映射

## 概述

Linux 支持多种优先级机制：

| 优先级类型 | 范围 | 适用调度策略 |
|-----------|------|-------------|
| 实时优先级 | 1-99 | SCHED_FIFO, SCHED_RR |
| 普通优先级 | 100-139 | SCHED_NORMAL, SCHED_BATCH |
| nice 值 | -20 到 +19 | 映射到普通优先级 |

## 一、优先级体系

### 优先级范围

```c
// include/linux/sched/prio.h
#define MAX_NICE        19
#define MIN_NICE        -20
#define NICE_WIDTH      (MAX_NICE - MIN_NICE + 1)

#define MAX_RT_PRIO     100
#define MAX_PRIO        (MAX_RT_PRIO + NICE_WIDTH)  // 140

// 优先级值越小，优先级越高
// 0-99:   实时优先级
// 100-139: 普通优先级（对应 nice -20 到 +19）
```

### 优先级关系图

```
优先级值    调度类      nice 值      说明
─────────────────────────────────────────────
   0        Stop        -          最高优先级
   1-99     RT          -          实时进程
  100       CFS        -20         普通进程最高
  120       CFS         0          普通进程默认
  139       CFS        +19         普通进程最低
─────────────────────────────────────────────
```

### task_struct 中的优先级字段

```c
// include/linux/sched.h
struct task_struct {
    // ...
    int                     prio;           // 动态优先级
    int                     static_prio;    // 静态优先级
    int                     normal_prio;    // 普通优先级
    unsigned int            rt_priority;    // 实时优先级 (1-99)
    
    unsigned int            policy;         // 调度策略
    // ...
};
```

### 优先级计算

```c
// kernel/sched/core.c

// nice 值转静态优先级
#define NICE_TO_PRIO(nice)  ((nice) + DEFAULT_PRIO)
// nice = 0 → prio = 120

// 静态优先级转 nice 值
#define PRIO_TO_NICE(prio)  ((prio) - DEFAULT_PRIO)
// prio = 120 → nice = 0

// 计算普通优先级
static inline int __normal_prio(struct task_struct *p)
{
    // 实时进程
    if (task_has_rt_policy(p))
        return MAX_RT_PRIO - 1 - p->rt_priority;
    
    // Deadline 进程
    if (task_has_dl_policy(p))
        return MAX_DL_PRIO - 1;
    
    // 普通进程
    return p->static_prio;
}

// 计算动态优先级
static int effective_prio(struct task_struct *p)
{
    // 实时进程使用固定优先级
    if (rt_prio(p->prio))
        return p->prio;
    
    // 普通进程可能受优先级继承影响
    return p->normal_prio;
}
```

## 二、nice 值

### nice 值概念

nice 值表示进程的"友好程度"：
- nice 值越大，进程越"友好"，优先级越低
- nice 值越小，进程越"不友好"，优先级越高

### 设置 nice 值

```c
// 系统调用
int nice(int inc);                    // 增加 nice 值
int setpriority(int which, id_t who, int prio);
int getpriority(int which, id_t who);

// 内核实现
// kernel/sys.c
SYSCALL_DEFINE1(nice, int, increment)
{
    long nice, retval;
    
    nice = task_nice(current) + increment;
    nice = clamp_val(nice, MIN_NICE, MAX_NICE);
    
    retval = security_task_setnice(current, nice);
    if (retval)
        return retval;
    
    set_user_nice(current, nice);
    return 0;
}
```

### set_user_nice() - 设置 nice 值

```c
// kernel/sched/core.c
void set_user_nice(struct task_struct *p, long nice)
{
    bool queued, running;
    int old_prio;
    struct rq *rq;
    
    if (task_nice(p) == nice)
        return;
    
    rq = task_rq_lock(p, &rf);
    
    queued = task_on_rq_queued(p);
    running = task_current(rq, p);
    
    // 如果在运行队列，先出队
    if (queued)
        dequeue_task(rq, p, DEQUEUE_SAVE | DEQUEUE_NOCLOCK);
    if (running)
        put_prev_task(rq, p);
    
    // 更新优先级
    old_prio = p->prio;
    p->static_prio = NICE_TO_PRIO(nice);
    p->prio = effective_prio(p);
    
    // 重新入队
    if (queued)
        enqueue_task(rq, p, ENQUEUE_RESTORE | ENQUEUE_NOCLOCK);
    if (running)
        set_next_task(rq, p);
    
    // 检查是否需要抢占
    if (queued) {
        resched_curr(rq);
    }
    
    task_rq_unlock(rq, p, &rf);
}
```

## 三、实时调度

### 实时调度策略

| 策略 | 说明 | 时间片 |
|-----|------|-------|
| SCHED_FIFO | 先进先出 | 无限（直到阻塞或让出） |
| SCHED_RR | 轮转 | 固定时间片（默认 100ms） |

### SCHED_FIFO

```c
// 特点：
// 1. 高优先级进程可以抢占低优先级进程
// 2. 同优先级进程不会相互抢占
// 3. 进程运行直到阻塞或主动让出

// 使用示例
struct sched_param param;
param.sched_priority = 50;  // 实时优先级 1-99
sched_setscheduler(pid, SCHED_FIFO, &param);
```

### SCHED_RR

```c
// 特点：
// 1. 高优先级进程可以抢占低优先级进程
// 2. 同优先级进程轮转运行
// 3. 时间片用完后排到队尾

// 使用示例
struct sched_param param;
param.sched_priority = 50;
sched_setscheduler(pid, SCHED_RR, &param);

// 查看时间片
int quantum = sched_rr_get_interval(pid, &tp);
```

### RT 调度器实现

```c
// kernel/sched/rt.c

// RT 优先级数组
struct rt_prio_array {
    DECLARE_BITMAP(bitmap, MAX_RT_PRIO + 1);  // 优先级位图
    struct list_head queue[MAX_RT_PRIO];       // 优先级链表
};

// 选择下一个 RT 任务
static struct task_struct *pick_next_task_rt(struct rq *rq)
{
    struct sched_rt_entity *rt_se;
    struct rt_rq *rt_rq = &rq->rt;
    
    // 找到最高优先级
    int idx = sched_find_first_bit(rt_rq->active.bitmap);
    if (idx >= MAX_RT_PRIO)
        return NULL;
    
    // 取队首任务
    rt_se = list_entry(rt_rq->active.queue[idx].next,
                       struct sched_rt_entity, run_list);
    
    return rt_task_of(rt_se);
}

// RR 时钟 tick
static void task_tick_rt(struct rq *rq, struct task_struct *p, int queued)
{
    update_curr_rt(rq);
    
    // SCHED_FIFO 不处理时间片
    if (p->policy != SCHED_RR)
        return;
    
    // RR 时间片递减
    if (--p->rt.time_slice)
        return;
    
    // 时间片用完，重置并移到队尾
    p->rt.time_slice = sched_rr_timeslice;
    
    if (p->rt.run_list.prev != p->rt.run_list.next) {
        requeue_task_rt(rq, p, 0);
        resched_curr(rq);
    }
}
```

### RT 带宽限制

```c
// 防止 RT 进程独占 CPU
// /proc/sys/kernel/sched_rt_period_us  = 1000000 (1秒)
// /proc/sys/kernel/sched_rt_runtime_us = 950000  (950ms)

// 每 1 秒周期内，RT 进程最多使用 950ms
// 剩余 50ms 给普通进程
```

## 四、Deadline 调度

### SCHED_DEADLINE 概念

Deadline 调度为任务提供确定性的时间保证：

```
参数：
- runtime: 每周期需要的 CPU 时间
- deadline: 相对截止时间
- period: 周期

调度保证：
每 period 时间内，任务获得 runtime 的 CPU 时间，
并在 deadline 之前完成
```

### 使用 Deadline 调度

```c
// 设置 Deadline 调度
struct sched_attr attr = {
    .size = sizeof(attr),
    .sched_policy = SCHED_DEADLINE,
    .sched_runtime = 10 * 1000 * 1000,   // 10ms
    .sched_deadline = 30 * 1000 * 1000,  // 30ms
    .sched_period = 100 * 1000 * 1000,   // 100ms
};
sched_setattr(pid, &attr, 0);

// 含义：每 100ms，任务获得 10ms CPU，在 30ms 内完成
```

### Deadline 调度器实现

```c
// kernel/sched/deadline.c

// Deadline 调度实体
struct sched_dl_entity {
    struct rb_node rb_node;     // 红黑树节点
    
    u64 dl_runtime;             // 运行时间配额
    u64 dl_deadline;            // 截止时间
    u64 dl_period;              // 周期
    
    s64 runtime;                // 剩余运行时间
    u64 deadline;               // 绝对截止时间
    
    unsigned int dl_throttled : 1;  // 是否被限流
    // ...
};

// 选择下一个 Deadline 任务（选择截止时间最早的）
static struct task_struct *pick_next_task_dl(struct rq *rq)
{
    struct sched_dl_entity *dl_se;
    struct dl_rq *dl_rq = &rq->dl;
    struct rb_node *node;
    
    if (!dl_rq->dl_nr_running)
        return NULL;
    
    // 取红黑树最左节点（deadline 最早）
    node = rb_first_cached(&dl_rq->root);
    dl_se = rb_entry(node, struct sched_dl_entity, rb_node);
    
    return dl_task_of(dl_se);
}
```

## 五、优先级继承

### 优先级反转问题

```
优先级反转场景：
1. 低优先级任务 L 持有锁 M
2. 高优先级任务 H 等待锁 M
3. 中优先级任务 M 抢占 L
4. H 被 M 间接阻塞（优先级反转）

       优先级
          │
    H ────┤ ★ 等待锁
          │
    M ────┤ ───────► 运行
          │
    L ────┤ ★ 持有锁
          └────────────► 时间
```

## 七、调度策略切换

### sched_setscheduler() - 设置调度策略

```c
// kernel/sched/core.c
SYSCALL_DEFINE3(sched_setscheduler, pid_t, pid, int, policy,
                struct sched_param __user *, param)
{
    return do_sched_setscheduler(pid, policy, param);
}

static int do_sched_setscheduler(pid_t pid, int policy,
                                 struct sched_param __user *param)
{
    struct sched_param lparam;
    struct task_struct *p;
    
    if (copy_from_user(&lparam, param, sizeof(struct sched_param)))
        return -EFAULT;
    
    p = find_process_by_pid(pid);
    if (!p)
        return -ESRCH;
    
    return sched_setscheduler(p, policy, &lparam);
}
```

## 总结

### 核心要点

1. **优先级体系**：
   - 实时优先级：1-99（值越大优先级越高）
   - 普通优先级：100-139（由 nice 值映射）

2. **nice 值**：
   - 范围：-20 到 +19
   - 值越小优先级越高
   - 影响 CFS 调度器的权重

3. **实时调度**：
   - SCHED_FIFO：先进先出，无时间片
   - SCHED_RR：轮转，固定时间片

4. **Deadline 调度**：
   - 提供确定性时间保证
   - 参数：runtime、deadline、period

5. **Android 优先级**：
   - 使用 OOM_ADJ 值
   - 映射到 Linux 优先级和 nice 值
