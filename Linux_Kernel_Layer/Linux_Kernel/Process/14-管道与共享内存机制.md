# 管道与共享内存机制

## 学习目标

- 理解匿名管道和命名管道的实现原理
- 掌握管道的创建和使用
- 理解共享内存的实现机制
- 了解 System V 和 POSIX 共享内存的区别
- 理解信号量与消息队列的基本概念

## 概述

管道和共享内存是 Linux 中两种重要的数据传输 IPC 机制：
- **管道**：通过内核缓冲区传输数据，适合流式数据
- **共享内存**：进程直接共享内存区域，效率最高

---

## 一、匿名管道（Pipe）

### 概念

匿名管道是最简单的 IPC 机制：
- 单向通信
- 仅限具有亲缘关系的进程
- 字节流，无消息边界

### 创建管道

```c
#include <unistd.h>

int fd[2];
int ret = pipe(fd);
// fd[0]: 读端
// fd[1]: 写端
```

### pipe() 系统调用实现

```c
// fs/pipe.c
SYSCALL_DEFINE1(pipe, int __user *, fildes)
{
    return do_pipe2(fildes, 0);
}

static int do_pipe2(int __user *fildes, int flags)
{
    struct file *files[2];
    int fd[2];
    int error;
    
    // 1. 创建管道
    error = __do_pipe_flags(fd, files, flags);
    if (error)
        return error;
    
    // 2. 复制文件描述符到用户空间
    if (copy_to_user(fildes, fd, sizeof(fd))) {
        put_unused_fd(fd[0]);
        put_unused_fd(fd[1]);
        fput(files[0]);
        fput(files[1]);
        return -EFAULT;
    }
    
    // 3. 安装文件描述符
    fd_install(fd[0], files[0]);
    fd_install(fd[1], files[1]);
    
    return 0;
}
```

### 管道结构

```c
// fs/pipe.c
struct pipe_inode_info {
    struct mutex mutex;
    wait_queue_head_t rd_wait, wr_wait;  // 读写等待队列
    unsigned int head;                    // 写入位置
    unsigned int tail;                    // 读取位置
    unsigned int max_usage;               // 最大缓冲区数
    unsigned int ring_size;               // 环形缓冲区大小
    unsigned int nr_accounted;
    unsigned int readers;                 // 读者数量
    unsigned int writers;                 // 写者数量
    unsigned int files;                   // 文件引用数
    unsigned int r_counter;               // 读计数
    unsigned int w_counter;               // 写计数
    struct page *tmp_page;
    struct fasync_struct *fasync_readers;
    struct fasync_struct *fasync_writers;
    struct pipe_buffer *bufs;             // 缓冲区数组
    struct user_struct *user;
};

struct pipe_buffer {
    struct page *page;      // 数据页
    unsigned int offset;    // 页内偏移
    unsigned int len;       // 数据长度
    const struct pipe_buf_operations *ops;
    unsigned int flags;
};
```

### 管道读写

```c
// fs/pipe.c
static ssize_t pipe_read(struct kiocb *iocb, struct iov_iter *to)
{
    struct file *filp = iocb->ki_filp;
    struct pipe_inode_info *pipe = filp->private_data;
    size_t total_len = iov_iter_count(to);
    int do_wakeup;
    ssize_t ret;
    
    __pipe_lock(pipe);
    
    for (;;) {
        unsigned int head = pipe->head;
        unsigned int tail = pipe->tail;
        unsigned int mask = pipe->ring_size - 1;
        
        // 检查是否有数据
        if (!pipe_empty(head, tail)) {
            struct pipe_buffer *buf = &pipe->bufs[tail & mask];
            size_t chars = buf->len;
            
            // 复制数据到用户空间
            if (copy_page_to_iter(buf->page, buf->offset, chars, to) != chars) {
                ret = -EFAULT;
                break;
            }
            
            pipe_buf_release(pipe, buf);
            pipe->tail = tail + 1;
            do_wakeup = 1;
            ret += chars;
            
            if (!total_len)
                break;
        }
        
        // 没有数据，检查写端是否关闭
        if (!pipe->writers)
            break;
        
        // 非阻塞模式
        if (filp->f_flags & O_NONBLOCK) {
            ret = -EAGAIN;
            break;
        }
        
        // 等待数据
        pipe_wait(pipe);
    }
    
    __pipe_unlock(pipe);
    
    if (do_wakeup)
        wake_up_interruptible_sync_poll(&pipe->wr_wait, ...);
    
    return ret;
}
```

### 管道使用示例

```c
int fd[2];
pipe(fd);

pid_t pid = fork();
if (pid == 0) {
    // 子进程：读端
    close(fd[1]);  // 关闭写端
    
    char buf[100];
    int n = read(fd[0], buf, sizeof(buf));
    printf("Child received: %s\n", buf);
    
    close(fd[0]);
} else {
    // 父进程：写端
    close(fd[0]);  // 关闭读端
    
    const char *msg = "Hello from parent";
    write(fd[1], msg, strlen(msg) + 1);
    
    close(fd[1]);
    wait(NULL);
}
```

## 二、命名管道（FIFO）

### 概念

命名管道解决了匿名管道只能用于亲缘进程的限制：
- 有文件系统路径名
- 任意进程可以打开
- 其他特性与匿名管道相同

### 创建 FIFO

```c
#include <sys/stat.h>

// 创建 FIFO
int mkfifo(const char *pathname, mode_t mode);

// 示例
mkfifo("/tmp/myfifo", 0666);
```

### mkfifo() 实现

```c
// fs/pipe.c
SYSCALL_DEFINE2(mkfifo, const char __user *, pathname, umode_t, mode)
{
    return sys_mknodat(AT_FDCWD, pathname, (mode & ~S_IFMT) | S_IFIFO, 0);
}
```

### FIFO 使用示例

```c
// 写进程
int fd = open("/tmp/myfifo", O_WRONLY);
write(fd, "Hello", 6);
close(fd);

// 读进程
int fd = open("/tmp/myfifo", O_RDONLY);
char buf[100];
read(fd, buf, sizeof(buf));
printf("Received: %s\n", buf);
close(fd);
```

## 三、共享内存

### 概念

共享内存是最高效的 IPC 机制：
- 数据不需要复制
- 需要同步机制
- 独立于进程生命周期

### System V 共享内存

```c
#include <sys/shm.h>

// 1. 创建/获取共享内存
int shmid = shmget(key, size, IPC_CREAT | 0666);

// 2. 附加到进程地址空间
void *ptr = shmat(shmid, NULL, 0);

// 3. 使用共享内存
memcpy(ptr, data, len);

// 4. 分离
shmdt(ptr);

// 5. 删除（可选）
shmctl(shmid, IPC_RMID, NULL);
```

### shmget() 实现

```c
// ipc/shm.c
SYSCALL_DEFINE3(shmget, key_t, key, size_t, size, int, shmflg)
{
    struct ipc_namespace *ns = current->nsproxy->ipc_ns;
    
    return ksys_shmget(ns, key, size, shmflg);
}

static long ksys_shmget(struct ipc_namespace *ns, key_t key,
                        size_t size, int shmflg)
{
    struct shmid_kernel *shp;
    int err;
    
    if (key == IPC_PRIVATE) {
        // 创建新的共享内存
        shp = newseg(ns, key, size, shmflg);
        if (IS_ERR(shp))
            return PTR_ERR(shp);
        return shp->shm_perm.id;
    }
    
    // 查找已存在的共享内存
    shp = shm_obtain_object_check(ns, key);
    if (IS_ERR(shp)) {
        if (shmflg & IPC_CREAT) {
            // 不存在，创建新的
            shp = newseg(ns, key, size, shmflg);
            if (IS_ERR(shp))
                return PTR_ERR(shp);
        } else {
            return PTR_ERR(shp);
        }
    }
    
    return shp->shm_perm.id;
}
```

### shmat() 实现

```c
// ipc/shm.c
SYSCALL_DEFINE3(shmat, int, shmid, char __user *, shmaddr, int, shmflg)
{
    unsigned long addr;
    long err;
    
    err = do_shmat(shmid, shmaddr, shmflg, &addr, SHMLBA);
    if (err)
        return err;
    
    return (long)addr;
}

long do_shmat(int shmid, char __user *shmaddr, int shmflg,
              unsigned long *raddr, unsigned long shmlba)
{
    struct shmid_kernel *shp;
    unsigned long addr;
    struct file *file;
    int err;
    
    // 1. 获取共享内存对象
    shp = shm_obtain_object_check(ns, shmid);
    if (IS_ERR(shp))
        return PTR_ERR(shp);
    
    // 2. 获取关联的文件
    file = shp->shm_file;
    get_file(file);
    
    // 3. 映射到进程地址空间
    addr = do_mmap(file, addr, size, prot, flags, 0, &uf);
    
    *raddr = addr;
    return 0;
}
```

### POSIX 共享内存

```c
#include <sys/mman.h>
#include <fcntl.h>

// 1. 创建/打开共享内存对象
int fd = shm_open("/myshm", O_CREAT | O_RDWR, 0666);

// 2. 设置大小
ftruncate(fd, 4096);

// 3. 映射
void *ptr = mmap(NULL, 4096, PROT_READ | PROT_WRITE, MAP_SHARED, fd, 0);

// 4. 使用
memcpy(ptr, data, len);

// 5. 取消映射
munmap(ptr, 4096);

// 6. 关闭
close(fd);

// 7. 删除
shm_unlink("/myshm");
```

### 共享内存同步

共享内存需要同步机制来防止竞态条件：

```c
// 使用信号量同步
sem_t *sem = sem_open("/mysem", O_CREAT, 0666, 1);

// 写操作
sem_wait(sem);
memcpy(shared_mem, data, len);
sem_post(sem);

// 读操作
sem_wait(sem);
memcpy(local_buf, shared_mem, len);
sem_post(sem);
```

## 四、信号量

### System V 信号量

```c
#include <sys/sem.h>

// 创建信号量集
int semid = semget(key, 1, IPC_CREAT | 0666);

// 初始化
union semun {
    int val;
    struct semid_ds *buf;
    unsigned short *array;
};
union semun arg;
arg.val = 1;
semctl(semid, 0, SETVAL, arg);

// P 操作（等待）
struct sembuf op = {0, -1, 0};
semop(semid, &op, 1);

// V 操作（释放）
struct sembuf op = {0, 1, 0};
semop(semid, &op, 1);

// 删除
semctl(semid, 0, IPC_RMID);
```

### POSIX 信号量

```c
#include <semaphore.h>

// 命名信号量
sem_t *sem = sem_open("/mysem", O_CREAT, 0666, 1);
sem_wait(sem);
sem_post(sem);
sem_close(sem);
sem_unlink("/mysem");

// 匿名信号量（用于线程或共享内存）
sem_t sem;
sem_init(&sem, 1, 1);  // 第二个参数 1 表示进程间共享
sem_wait(&sem);
sem_post(&sem);
sem_destroy(&sem);
```

## 五、消息队列

### System V 消息队列

```c
#include <sys/msg.h>

// 消息结构
struct msgbuf {
    long mtype;     // 消息类型
    char mtext[256]; // 消息内容
};

// 创建消息队列
int msgid = msgget(key, IPC_CREAT | 0666);

// 发送消息
struct msgbuf msg = {.mtype = 1, .mtext = "Hello"};
msgsnd(msgid, &msg, strlen(msg.mtext) + 1, 0);

// 接收消息
struct msgbuf msg;
msgrcv(msgid, &msg, sizeof(msg.mtext), 1, 0);
// 第四个参数是消息类型，0 表示接收任意类型

// 删除
msgctl(msgid, IPC_RMID, NULL);
```

### POSIX 消息队列

```c
#include <mqueue.h>

// 创建消息队列
struct mq_attr attr = {
    .mq_maxmsg = 10,
    .mq_msgsize = 256,
};
mqd_t mq = mq_open("/myqueue", O_CREAT | O_RDWR, 0666, &attr);

// 发送消息
mq_send(mq, "Hello", 6, 0);  // 优先级 0

// 接收消息
char buf[256];
unsigned int prio;
mq_receive(mq, buf, sizeof(buf), &prio);

// 关闭和删除
mq_close(mq);
mq_unlink("/myqueue");
```

## 六、对比总结

### 管道 vs 共享内存

| 特性 | 管道 | 共享内存 |
|-----|------|---------|
| 数据复制 | 2次 | 0次 |
| 效率 | 中 | 高 |
| 同步 | 内置 | 需自行实现 |
| 方向 | 单向 | 双向 |
| 适用场景 | 流式数据 | 大量数据交换 |

### System V vs POSIX

| 特性 | System V | POSIX |
|-----|----------|-------|
| 命名 | 整数 key | 字符串名称 |
| API | 复杂 | 简洁 |
| 移植性 | 广泛 | 需要 librt |
| 资源管理 | 手动 | 类文件系统 |

---

## 总结

### 核心要点

1. **管道**：
   - 匿名管道：`pipe()`，父子进程
   - 命名管道：`mkfifo()`，任意进程

2. **共享内存**：
   - 效率最高，零拷贝
   - 需要同步机制

3. **信号量**：
   - 同步机制
   - P/V 操作

4. **消息队列**：
   - 消息有边界
   - 支持优先级
