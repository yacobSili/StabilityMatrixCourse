# 输入设备驱动与事件上报

## 引言

本文深入分析 Android 设备中常见的输入设备驱动，包括 GPIO 按键驱动和触摸屏驱动，理解它们如何与 Input 子系统交互并上报事件。

---

## 1. GPIO 按键驱动 (gpio-keys)

### 1.1 驱动架构

gpio-keys 是 Linux 内核提供的通用 GPIO 按键驱动，适用于通过 GPIO 连接的独立按键。

```
┌─────────────────────────────────────────────────────────────────┐
│                     gpio-keys 驱动                               │
│  ┌───────────────────────────────────────────────────────────┐  │
│  │  gpio_keys_probe()                                        │  │
│  │  - 解析 DTS 配置                                          │  │
│  │  - 申请 GPIO                                              │  │
│  │  - 注册 input_dev                                         │  │
│  │  - 注册中断处理                                           │  │
│  └───────────────────────────────────────────────────────────┘  │
│                                                                  │
│  ┌───────────────────────────────────────────────────────────┐  │
│  │  gpio_keys_irq_isr() / gpio_keys_gpio_isr()               │  │
│  │  - 中断服务程序                                           │  │
│  │  - 读取 GPIO 状态                                         │  │
│  │  - 消抖处理                                               │  │
│  │  - input_report_key() + input_sync()                     │  │
│  └───────────────────────────────────────────────────────────┘  │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│                        Input Core                                │
│                 input_event() → evdev_event()                   │
└─────────────────────────────────────────────────────────────────┘
```

### 1.3 核心代码分析

```c
// drivers/input/keyboard/gpio_keys.c

struct gpio_button_data {
    const struct gpio_keys_button *button;
    struct input_dev *input;
    struct gpio_desc *gpiod;
    
    unsigned int irq;
    spinlock_t lock;
    bool disabled;
    bool key_pressed;
    bool suspended;
    
    struct delayed_work work;           // 消抖延时工作
    unsigned int software_debounce;     // 软件消抖时间
};

struct gpio_keys_drvdata {
    const struct gpio_keys_platform_data *pdata;
    struct input_dev *input;
    struct mutex disable_lock;
    unsigned short *keymap;
    struct gpio_button_data data[];     // 按键数据数组
};

// probe 函数 - 驱动初始化
static int gpio_keys_probe(struct platform_device *pdev)
{
    struct device *dev = &pdev->dev;
    const struct gpio_keys_platform_data *pdata;
    struct gpio_keys_drvdata *ddata;
    struct input_dev *input;
    int i, error;
    
    // 获取平台数据或解析设备树
    pdata = gpio_keys_get_devtree_pdata(dev);
    
    // 分配驱动数据
    ddata = devm_kzalloc(dev, struct_size(ddata, data, pdata->nbuttons),
                         GFP_KERNEL);
    
    // 分配 input_dev
    input = devm_input_allocate_device(dev);
    
    ddata->pdata = pdata;
    ddata->input = input;
    
    // 设置 input_dev 属性
    input->name = pdata->name ? : pdev->name;
    input->phys = "gpio-keys/input0";
    input->dev.parent = dev;
    
    input->id.bustype = BUS_HOST;
    input->id.vendor = 0x0001;
    input->id.product = 0x0001;
    input->id.version = 0x0100;
    
    // 设置设备能力
    __set_bit(EV_KEY, input->evbit);
    if (pdata->rep)
        __set_bit(EV_REP, input->evbit);
    
    // 设置每个按键
    for (i = 0; i < pdata->nbuttons; i++) {
        const struct gpio_keys_button *button = &pdata->buttons[i];
        struct gpio_button_data *bdata = &ddata->data[i];
        
        error = gpio_keys_setup_key(pdev, input, ddata, button, i, NULL);
        if (error)
            return error;
        
        // 设置 keybit
        __set_bit(button->code, input->keybit);
    }
    
    // 注册 input_dev
    error = input_register_device(input);
    
    return 0;
}

// 设置单个按键
static int gpio_keys_setup_key(struct platform_device *pdev,
                               struct input_dev *input,
                               struct gpio_keys_drvdata *ddata,
                               const struct gpio_keys_button *button,
                               int idx,
                               struct fwnode_handle *child)
{
    struct gpio_button_data *bdata = &ddata->data[idx];
    unsigned long irqflags;
    int error;
    
    bdata->input = input;
    bdata->button = button;
    
    // 获取 GPIO
    bdata->gpiod = devm_fwnode_gpiod_get(dev, child, NULL, GPIOD_IN, desc);
    
    // 设置消抖
    if (button->debounce_interval) {
        error = gpiod_set_debounce(bdata->gpiod, 
                                   button->debounce_interval * 1000);
        if (error < 0)
            bdata->software_debounce = button->debounce_interval;
    }
    
    // 获取中断号
    bdata->irq = gpiod_to_irq(bdata->gpiod);
    
    // 初始化消抖工作队列
    INIT_DELAYED_WORK(&bdata->work, gpio_keys_gpio_work_func);
    
    // 申请中断
    irqflags = IRQF_TRIGGER_RISING | IRQF_TRIGGER_FALLING;
    error = devm_request_any_context_irq(dev, bdata->irq,
                                         gpio_keys_gpio_isr, irqflags,
                                         desc, bdata);
    
    return 0;
}

// GPIO 中断服务程序
static irqreturn_t gpio_keys_gpio_isr(int irq, void *dev_id)
{
    struct gpio_button_data *bdata = dev_id;
    
    // 使用延时工作队列进行消抖
    mod_delayed_work(system_wq, &bdata->work,
                     msecs_to_jiffies(bdata->software_debounce));
    
    return IRQ_HANDLED;
}

// 消抖后的工作函数
static void gpio_keys_gpio_work_func(struct work_struct *work)
{
    struct gpio_button_data *bdata =
        container_of(work, struct gpio_button_data, work.work);
    
    gpio_keys_gpio_report_event(bdata);
}

// 上报按键事件
static void gpio_keys_gpio_report_event(struct gpio_button_data *bdata)
{
    const struct gpio_keys_button *button = bdata->button;
    struct input_dev *input = bdata->input;
    unsigned int type = button->type ?: EV_KEY;
    int state;
    
    // 读取 GPIO 状态
    state = gpiod_get_value_cansleep(bdata->gpiod);
    
    // 处理 active_low
    if (state < 0) {
        dev_err(input->dev.parent, "failed to get gpio state: %d\n", state);
        return;
    }
    
    // 上报事件
    if (type == EV_ABS) {
        if (state)
            input_event(input, type, button->code, button->value);
    } else {
        // 按键事件
        input_event(input, type, button->code, state);
    }
    
    // 同步
    input_sync(input);
}
```

### 2.2 通用触摸屏驱动框架

以一个简化的触摸屏驱动为例：

```c
// 触摸点数据结构
struct touch_point {
    int id;             // 追踪 ID
    int x;              // X 坐标
    int y;              // Y 坐标
    int pressure;       // 压力
    int touch_major;    // 触摸区域长轴
    bool active;        // 是否有效
};

// 触摸屏驱动数据
struct touchscreen_data {
    struct i2c_client *client;
    struct input_dev *input;
    
    int max_x;
    int max_y;
    int max_touch_points;
    
    struct touch_point points[MAX_TOUCH_POINTS];
    
    struct gpio_desc *reset_gpio;
    struct gpio_desc *irq_gpio;
    
    bool suspended;
};

// probe 函数
static int touchscreen_probe(struct i2c_client *client)
{
    struct device *dev = &client->dev;
    struct touchscreen_data *ts;
    struct input_dev *input;
    int error;
    
    // 分配驱动数据
    ts = devm_kzalloc(dev, sizeof(*ts), GFP_KERNEL);
    ts->client = client;
    
    // 获取设备树属性
    device_property_read_u32(dev, "touchscreen-size-x", &ts->max_x);
    device_property_read_u32(dev, "touchscreen-size-y", &ts->max_y);
    device_property_read_u32(dev, "touchscreen-max-points", 
                             &ts->max_touch_points);
    
    // 获取 GPIO
    ts->reset_gpio = devm_gpiod_get_optional(dev, "reset", GPIOD_OUT_HIGH);
    ts->irq_gpio = devm_gpiod_get_optional(dev, "irq", GPIOD_IN);
    
    // 硬件初始化
    touchscreen_hw_init(ts);
    
    // 分配并配置 input_dev
    input = devm_input_allocate_device(dev);
    ts->input = input;
    
    input->name = "Generic Touchscreen";
    input->id.bustype = BUS_I2C;
    input->dev.parent = dev;
    
    // 设置事件能力
    __set_bit(EV_ABS, input->evbit);
    __set_bit(EV_KEY, input->evbit);
    __set_bit(BTN_TOUCH, input->keybit);
    
    // 设置绝对坐标轴
    input_set_abs_params(input, ABS_X, 0, ts->max_x, 0, 0);
    input_set_abs_params(input, ABS_Y, 0, ts->max_y, 0, 0);
    input_set_abs_params(input, ABS_PRESSURE, 0, 255, 0, 0);
    
    // 设置多点触控
    input_mt_init_slots(input, ts->max_touch_points, 
                        INPUT_MT_DIRECT | INPUT_MT_DROP_UNUSED);
    input_set_abs_params(input, ABS_MT_POSITION_X, 0, ts->max_x, 0, 0);
    input_set_abs_params(input, ABS_MT_POSITION_Y, 0, ts->max_y, 0, 0);
    input_set_abs_params(input, ABS_MT_PRESSURE, 0, 255, 0, 0);
    input_set_abs_params(input, ABS_MT_TOUCH_MAJOR, 0, 255, 0, 0);
    
    // 注册中断
    error = devm_request_threaded_irq(dev, client->irq,
                                      NULL, touchscreen_irq_handler,
                                      IRQF_ONESHOT | IRQF_TRIGGER_FALLING,
                                      "touchscreen", ts);
    
    // 注册 input_dev
    error = input_register_device(input);
    
    return 0;
}

// 中断处理函数 (线程化中断)
static irqreturn_t touchscreen_irq_handler(int irq, void *dev_id)
{
    struct touchscreen_data *ts = dev_id;
    struct touch_point points[MAX_TOUCH_POINTS];
    int touch_count;
    int i;
    
    // 读取触摸数据
    touch_count = touchscreen_read_data(ts, points);
    if (touch_count < 0) {
        dev_err(&ts->client->dev, "Failed to read touch data\n");
        return IRQ_HANDLED;
    }
    
    // 上报多点触控事件
    for (i = 0; i < ts->max_touch_points; i++) {
        struct touch_point *p = &points[i];
        
        // 选择槽位
        input_mt_slot(ts->input, i);
        
        if (p->active) {
            // 上报触摸点
            input_mt_report_slot_state(ts->input, MT_TOOL_FINGER, true);
            input_report_abs(ts->input, ABS_MT_POSITION_X, p->x);
            input_report_abs(ts->input, ABS_MT_POSITION_Y, p->y);
            input_report_abs(ts->input, ABS_MT_PRESSURE, p->pressure);
            input_report_abs(ts->input, ABS_MT_TOUCH_MAJOR, p->touch_major);
        } else {
            // 触摸点释放
            input_mt_report_slot_state(ts->input, MT_TOOL_FINGER, false);
        }
    }
    
    // 同步多点触控帧
    input_mt_sync_frame(ts->input);
    
    // 同步单点触控 (向后兼容)
    input_mt_report_pointer_emulation(ts->input, true);
    
    // 完成事件上报
    input_sync(ts->input);
    
    return IRQ_HANDLED;
}

// I2C 读取触摸数据
static int touchscreen_read_data(struct touchscreen_data *ts,
                                 struct touch_point *points)
{
    u8 buf[64];
    int ret;
    int i, touch_count = 0;
    
    // I2C 读取
    ret = i2c_master_recv(ts->client, buf, sizeof(buf));
    if (ret < 0)
        return ret;
    
    // 解析数据 (协议因厂商而异)
    touch_count = buf[0] & 0x0F;
    
    for (i = 0; i < ts->max_touch_points; i++) {
        int offset = 1 + i * 6;
        u8 event = buf[offset] >> 6;
        
        points[i].id = i;
        
        if (event == 0 || event == 2) {  // Press or Contact
            points[i].active = true;
            points[i].x = ((buf[offset] & 0x0F) << 8) | buf[offset + 1];
            points[i].y = ((buf[offset + 2] & 0x0F) << 8) | buf[offset + 3];
            points[i].pressure = buf[offset + 4];
            points[i].touch_major = buf[offset + 5];
        } else {
            points[i].active = false;
        }
    }
    
    return touch_count;
}
```

## 3. Input 设备注册流程

### 3.1 注册流程图

```
驱动 probe()
    │
    ▼
┌──────────────────────────────────────┐
│ input = devm_input_allocate_device() │
│ - 分配 input_dev 结构                │
│ - 初始化链表、锁等                   │
└──────────────────────┬───────────────┘
                       │
                       ▼
┌──────────────────────────────────────┐
│ 配置 input_dev                       │
│ - 设置 name, phys, id               │
```

## 4. 事件上报最佳实践

### 4.1 按键事件上报

```c
// 按键按下
input_report_key(input, KEY_POWER, 1);
input_sync(input);

// 按键释放
input_report_key(input, KEY_POWER, 0);
input_sync(input);

// 注意: 
// - value = 0 表示释放, value = 1 表示按下, value = 2 表示重复
// - 每次状态变化都要 input_sync()
```

### 4.2 单点触摸上报

```c
// 触摸按下
input_report_key(input, BTN_TOUCH, 1);
input_report_abs(input, ABS_X, x);
input_report_abs(input, ABS_Y, y);
input_report_abs(input, ABS_PRESSURE, pressure);
input_sync(input);

// 触摸移动
input_report_abs(input, ABS_X, new_x);
input_report_abs(input, ABS_Y, new_y);
input_sync(input);

// 触摸释放
input_report_key(input, BTN_TOUCH, 0);
input_report_abs(input, ABS_PRESSURE, 0);
input_sync(input);
```

### 4.3 多点触控上报 (Type B)

```c
// 报告多个触摸点
for (i = 0; i < touch_count; i++) {
    // 选择槽位
    input_mt_slot(input, i);
    
    // 报告触摸状态
    input_mt_report_slot_state(input, MT_TOOL_FINGER, points[i].active);
    
    if (points[i].active) {
        input_report_abs(input, ABS_MT_POSITION_X, points[i].x);
        input_report_abs(input, ABS_MT_POSITION_Y, points[i].y);
        input_report_abs(input, ABS_MT_PRESSURE, points[i].pressure);
        input_report_abs(input, ABS_MT_TOUCH_MAJOR, points[i].major);
    }
}

// 同步多点触控帧 (自动处理未更新的槽位)
input_mt_sync_frame(input);

// 生成单点触控兼容事件
input_mt_report_pointer_emulation(input, true);

// 完成帧
input_sync(input);
```

## 6. 本章小结

本文介绍了 Android 常见输入设备驱动的实现：

| 驱动类型 | 关键技术点 |
|---------|----------|
| GPIO 按键 | 中断处理、消抖、GPIO 子系统 |
| 触摸屏 | I2C 通信、MT 协议、坐标转换 |

事件上报要点：
1. 正确设置 input_dev 能力位
2. 使用 Type B 多点触控协议
3. 每帧调用 input_sync()
4. 使用 input_mt_sync_frame() 处理释放的触摸点
