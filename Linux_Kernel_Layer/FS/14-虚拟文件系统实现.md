# 虚拟文件系统实现

## 学习目标

- 理解虚拟文件系统的概念和实现模式
- 掌握 procfs：进程信息接口
- 理解 sysfs：设备模型接口
- 了解 debugfs：调试接口
- 理解 devtmpfs：设备文件管理

## 概述

虚拟文件系统（Virtual File System）不存储实际数据，而是提供内核信息的接口。它们在 Linux 和 Android 系统中发挥着重要作用。

---

## 一、虚拟文件系统概述

### 虚拟文件系统的特点

1. **不存储数据**：数据动态生成
2. **只读或读写**：根据用途决定
3. **内核接口**：提供内核信息访问
4. **特殊操作**：自定义 read/write 实现

### 常见虚拟文件系统

| 文件系统 | 挂载点 | 用途 |
|---------|-------|-----|
| procfs | /proc | 进程和系统信息 |
| sysfs | /sys | 设备模型信息 |
| debugfs | /sys/kernel/debug | 调试信息 |
| devtmpfs | /dev | 设备文件管理 |
| tmpfs | /tmp, /dev/shm | 临时文件系统 |

## 二、procfs（/proc）

### procfs 的作用

procfs 提供进程和系统信息的接口，是用户空间了解内核状态的重要途径。

### procfs 实现

```c
// fs/proc/root.c
static struct file_system_type proc_fs_type = {
    .name       = "proc",
    .mount      = proc_mount,
    .kill_sb    = proc_kill_sb,
    .fs_flags   = FS_USERNS_MOUNT,
};

static int __init init_proc_fs(void)
{
    int err = proc_init_inodecache();
    if (err)
        goto out;
    err = register_filesystem(&proc_fs_type);
    if (err)
        goto out;
    return 0;
}
```

### proc 文件操作

```c
// fs/proc/inode.c
static const struct file_operations proc_file_operations = {
    .llseek		= proc_file_lseek,
    .read		= proc_file_read,
    .write		= proc_file_write,
    .poll		= proc_file_poll,
    .unlocked_ioctl = proc_file_unlocked_ioctl,
    .mmap		= proc_file_mmap,
    .open		= proc_file_open,
    .release	= proc_file_release,
};
```

### 常见 proc 文件

```bash
# 进程信息
/proc/<pid>/stat      # 进程状态
/proc/<pid>/status     # 进程详细信息
/proc/<pid>/maps       # 内存映射
/proc/<pid>/fd/        # 文件描述符

# 系统信息
/proc/meminfo          # 内存信息
/proc/cpuinfo          # CPU信息
/proc/version          # 内核版本
/proc/mounts           # 挂载信息
```

## 三、sysfs（/sys）

### sysfs 的作用

sysfs 提供设备模型信息，是 Linux 设备模型的核心接口。

### sysfs 实现

```c
// fs/sysfs/mount.c
static struct file_system_type sysfs_fs_type = {
    .name       = "sysfs",
    .mount      = sysfs_mount,
    .kill_sb    = sysfs_kill_sb,
    .fs_flags   = FS_USERNS_MOUNT | FS_USERNS_VISIBLE,
};
```

### sysfs 目录结构

```
/sys
├── block/          # 块设备
├── bus/            # 总线
├── class/          # 设备类
├── dev/            # 设备节点
├── devices/        # 设备树
└── kernel/         # 内核信息
```

### sysfs 文件操作

```c
// fs/sysfs/file.c
static const struct file_operations sysfs_file_operations = {
    .read       = sysfs_read_file,
    .write      = sysfs_write_file,
    .llseek     = generic_file_llseek,
    .open       = sysfs_open_file,
    .release    = sysfs_release,
    .poll       = sysfs_poll,
};
```

## 四、debugfs（/sys/kernel/debug）

### debugfs 的作用

debugfs 提供调试信息接口，用于内核开发和调试。

### debugfs 实现

```c
// fs/debugfs/inode.c
static struct file_system_type debug_fs_type = {
    .name = "debugfs",
    .mount = debug_mount,
    .kill_sb = kill_litter_super,
};
```

### debugfs 使用示例

```c
// 内核代码中创建 debugfs 文件
struct dentry *debugfs_create_file(const char *name, umode_t mode,
                                   struct dentry *parent, void *data,
                                   const struct file_operations *fops)
{
    struct dentry *dentry;
    struct inode *inode;
    
    dentry = start_creating(name, parent);
    if (IS_ERR(dentry))
        return dentry;
    
    inode = debugfs_get_inode(dentry->d_sb);
    inode->i_mode = mode;
    inode->i_fop = fops ? &debugfs_full_proxy_file_operations :
                          &debugfs_open_proxy_file_operations;
    
    d_instantiate(dentry, inode);
    return dentry;
}
```

## 五、devtmpfs（/dev）

### devtmpfs 的作用

devtmpfs 管理设备文件，在系统启动早期提供设备文件访问。

### devtmpfs 实现

```c
// drivers/base/devtmpfs.c
static struct file_system_type dev_fs_type = {
    .name = "devtmpfs",
    .mount = dev_mount,
    .kill_sb = kill_litter_super,
};
```

### 设备文件创建

```c
// drivers/base/devtmpfs.c
int devtmpfs_create_node(struct device *dev)
{
    const char *tmp = NULL;
    const char *nodename;
    struct dentry *dentry;
    struct inode *inode;
    int mode = 0;
    
    // 1. 确定设备文件名
    nodename = device_get_devnode(dev, &mode, &tmp, NULL);
    
    // 2. 创建设备文件
    dentry = kern_path_create(AT_FDCWD, nodename, &path, LOOKUP_PARENT);
    if (IS_ERR(dentry))
        return PTR_ERR(dentry);
    
    // 3. 创建 inode
    inode = new_inode(dev_mnt->mnt_sb);
    inode->i_mode = mode;
    inode->i_uid = GLOBAL_ROOT_UID;
    inode->i_gid = GLOBAL_ROOT_GID;
    inode->i_fop = &def_blk_fops;  // 或 def_chr_fops
    
    d_instantiate(dentry, inode);
    
    return 0;
}
```

## 六、虚拟文件系统的实现模式

### 通用实现模式

```c
// 1. 定义文件系统类型
static struct file_system_type my_vfs_type = {
    .name = "myvfs",
    .mount = my_vfs_mount,
    .kill_sb = my_vfs_kill_sb,
};

// 2. 实现文件操作
static const struct file_operations my_vfs_file_operations = {
    .read = my_vfs_read,
    .write = my_vfs_write,
    .open = my_vfs_open,
    .release = my_vfs_release,
};

// 3. 实现 inode 操作
static const struct inode_operations my_vfs_inode_operations = {
    .lookup = my_vfs_lookup,
    .getattr = my_vfs_getattr,
};

// 4. 数据生成函数
static ssize_t my_vfs_read(struct file *file, char __user *buf,
                           size_t count, loff_t *ppos)
{
    // 动态生成数据
    char data[256];
    snprintf(data, sizeof(data), "Dynamic data: %ld\n", jiffies);
    
    // 复制到用户空间
    return simple_read_from_buffer(buf, count, ppos, data, strlen(data));
}
```
