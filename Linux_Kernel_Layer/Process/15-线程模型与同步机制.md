# 线程模型与同步机制

## 学习目标

- 理解 Linux 线程实现（NPTL）
- 掌握线程与进程的关系
- 理解 pthread_create 与 clone() 的关系
- 掌握各种同步机制：互斥锁、条件变量、自旋锁、读写锁

## 概述

Linux 将线程实现为轻量级进程（LWP），共享地址空间和资源。线程是 CPU 调度的基本单位，进程是资源分配的基本单位。

## 一、Linux 线程实现

### NPTL（Native POSIX Threads Library）

NPTL 是 Linux 的标准线程库，特点：
- 1:1 线程模型（用户线程对应内核线程）
- 基于 clone() 系统调用
- 完整的 POSIX 线程语义

### 线程与进程的关系

```c
// 进程：独立的资源集合
struct task_struct {
    pid_t pid;              // 进程 ID
    pid_t tgid;             // 线程组 ID
    struct mm_struct *mm;   // 地址空间
    struct files_struct *files; // 文件
    // ...
};

// 线程：同一线程组内的多个 task_struct
// 主线程：pid == tgid
// 子线程：pid != tgid, tgid 等于主线程的 pid
```

### 线程组示意图

```
进程（线程组）
┌─────────────────────────────────────────────────────────────┐
│  tgid = 1000                                                │
│                                                              │
│  ┌──────────────┐  ┌──────────────┐  ┌──────────────┐      │
│  │ 主线程       │  │ 工作线程 1   │  │ 工作线程 2   │      │
│  │ pid = 1000   │  │ pid = 1001   │  │ pid = 1002   │      │
│  │ tgid = 1000  │  │ tgid = 1000  │  │ tgid = 1000  │      │
│  └──────────────┘  └──────────────┘  └──────────────┘      │
│                                                              │
│  共享：                                                      │
│  - mm_struct（地址空间）                                    │

│  - fs_struct（文件系统信息）                                │
│  - signal_struct（信号结构）                                │
│                                                              │
│  独立：                                                      │
│  - task_struct                                              │
│  - 内核栈                                                   │
│  - 线程局部存储（TLS）                                      │
└─────────────────────────────────────────────────────────────┘
```

## 二、线程创建

### pthread_create()

```c
#include <pthread.h>

int pthread_create(pthread_t *thread,
                   const pthread_attr_t *attr,
                   void *(*start_routine)(void *),
                   void *arg);
```

### pthread_create 内部实现

```c
// glibc/nptl/pthread_create.c
int __pthread_create_2_1(pthread_t *newthread,
                         const pthread_attr_t *attr,
                         void *(*start_routine)(void *),
                         void *arg)
{
    struct pthread *pd;
    int err;
    
    // 1. 分配线程控制块和栈
    err = ALLOCATE_STACK(iattr, &pd);
    if (err)
        return err;
    
    // 2. 初始化线程控制块
    pd->start_routine = start_routine;
    pd->arg = arg;
    
    // 3. 调用 clone() 创建线程
    const int clone_flags = (CLONE_VM | CLONE_FS | CLONE_FILES | CLONE_SIGHAND
                            | CLONE_THREAD | CLONE_SYSVSEM | CLONE_SETTLS
                            | CLONE_PARENT_SETTID | CLONE_CHILD_CLEARTID);
    
    err = ARCH_CLONE(&start_thread, STACK_VARIABLES_ARGS, clone_flags,
                     pd, &pd->tid, tp, &pd->tid);
    
    *newthread = (pthread_t)pd;
    return 0;
}
```

### clone() 标志详解

```c
// 线程创建使用的标志
CLONE_VM        // 共享地址空间
CLONE_FS        // 共享文件系统信息
CLONE_FILES     // 共享文件描述符表
CLONE_SIGHAND   // 共享信号处理函数
CLONE_THREAD    // 同一线程组
CLONE_SYSVSEM   // 共享 System V 信号量
CLONE_SETTLS    // 设置线程局部存储
CLONE_PARENT_SETTID  // 在父线程设置 TID
CLONE_CHILD_CLEARTID // 在子线程退出时清除 TID
```

### clone() 处理 CLONE_THREAD

```c
// kernel/fork.c
static __latent_entropy struct task_struct *copy_process(...)
{
    // ...
    
    if (clone_flags & CLONE_THREAD) {
        // 线程：加入同一线程组
        p->tgid = current->tgid;
        p->group_leader = current->group_leader;
        list_add_tail_rcu(&p->thread_group, &p->group_leader->thread_group);
    } else {
        // 进程：创建新线程组
        p->tgid = p->pid;
        p->group_leader = p;
        INIT_LIST_HEAD(&p->thread_group);
    }
    
    // ...
}
```

## 三、线程局部存储（TLS）

### 概念

TLS 允许每个线程拥有变量的独立副本。

### 使用方式

```c
// 1. __thread 关键字（GCC 扩展）
__thread int thread_local_var;

// 2. pthread_key_create
pthread_key_t key;
pthread_key_create(&key, destructor);
pthread_setspecific(key, value);
void *value = pthread_getspecific(key);

// 3. C11 thread_local
_Thread_local int thread_local_var;
```

### TLS 实现

```c
// 每个线程有独立的 TLS 区域
// ARM64: tp (thread pointer) 寄存器指向 TLS

// clone() 时设置 TLS
// arch/arm64/kernel/process.c
int copy_thread(...)
{
    if (clone_flags & CLONE_SETTLS)
        p->thread.uw.tp_value = tls;
}
```

## 四、互斥锁（Mutex）

### pthread_mutex

```c
pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;

// 或动态初始化
pthread_mutex_init(&mutex, NULL);

// 加锁
pthread_mutex_lock(&mutex);

// 尝试加锁（非阻塞）
int ret = pthread_mutex_trylock(&mutex);

// 解锁
pthread_mutex_unlock(&mutex);

// 销毁
pthread_mutex_destroy(&mutex);
```

### Futex（Fast Userspace Mutex）

互斥锁的内核实现基于 futex：

```c
// include/linux/futex.h
// Futex 的核心思想：
// - 无竞争时完全在用户空间操作
// - 有竞争时才进入内核等待

// 用户空间加锁（无竞争）
if (atomic_cmpxchg(&mutex, 0, 1) == 0) {
    // 成功获取锁，不需要进入内核
} else {
    // 有竞争，进入内核等待
    futex(FUTEX_WAIT, &mutex, 1);
}

// 用户空间解锁
if (atomic_cmpxchg(&mutex, 1, 0) == 1) {
    // 没有等待者
} else {
    // 有等待者，唤醒
    futex(FUTEX_WAKE, &mutex, 1);
}
```

### futex() 系统调用

```c
// kernel/futex.c
SYSCALL_DEFINE6(futex, u32 __user *, uaddr, int, op, u32, val,
                const struct __kernel_timespec __user *, utime,
                u32 __user *, uaddr2, u32, val3)
{
    switch (op & FUTEX_CMD_MASK) {
    case FUTEX_WAIT:
        return futex_wait(uaddr, flags, val, timeout);
    case FUTEX_WAKE:
        return futex_wake(uaddr, flags, val);
    // ...
    }
}

// 等待
static int futex_wait(u32 __user *uaddr, unsigned int flags,
                      u32 val, ktime_t *timeout)
{
    struct futex_hash_bucket *hb;
    struct futex_q q;
    
    // 1. 检查值是否匹配
    if (get_futex_value_locked(&uval, uaddr))
        return -EFAULT;
    if (uval != val)
        return -EAGAIN;
    
    // 2. 加入等待队列
    futex_queue(&q, hb);
    
    // 3. 睡眠
    if (!timeout)
        schedule();
    else
        schedule_hrtimeout_range_clock(timeout, 0, clock, HRTIMER_MODE_ABS);
    
    return 0;
}
```

## 五、条件变量

### pthread_cond

```c
pthread_cond_t cond = PTHREAD_COND_INITIALIZER;
pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;

// 生产者
pthread_mutex_lock(&mutex);
// 生产数据
data_ready = 1;
pthread_cond_signal(&cond);  // 通知一个等待者
// 或 pthread_cond_broadcast(&cond);  // 通知所有等待者
pthread_mutex_unlock(&mutex);

// 消费者
pthread_mutex_lock(&mutex);
while (!data_ready) {
    pthread_cond_wait(&cond, &mutex);  // 等待条件
}
// 消费数据
pthread_mutex_unlock(&mutex);
```

## 六、自旋锁（Spinlock）

### 概念

自旋锁在等待时忙等待（不睡眠）：
- 适用于短时间持有的锁
- 不能在持有时睡眠
- 主要用于内核

### pthread_spinlock

```c
pthread_spinlock_t spinlock;

pthread_spin_init(&spinlock, PTHREAD_PROCESS_PRIVATE);
pthread_spin_lock(&spinlock);
// 临界区
pthread_spin_unlock(&spinlock);
pthread_spin_destroy(&spinlock);
```

### 内核自旋锁

```c
// include/linux/spinlock.h
spinlock_t lock;

spin_lock_init(&lock);
spin_lock(&lock);
// 临界区（不能睡眠）
spin_unlock(&lock);

// 禁用中断版本
spin_lock_irqsave(&lock, flags);
spin_unlock_irqrestore(&lock, flags);
```

## 七、读写锁（RWLock）

### 概念

读写锁允许多个读者同时访问，但写者独占：

```
读者-读者：并发
读者-写者：互斥
写者-写者：互斥
```

### pthread_rwlock

```c
pthread_rwlock_t rwlock = PTHREAD_RWLOCK_INITIALIZER;

// 读锁
pthread_rwlock_rdlock(&rwlock);
// 读操作
pthread_rwlock_unlock(&rwlock);

// 写锁
pthread_rwlock_wrlock(&rwlock);
// 写操作
pthread_rwlock_unlock(&rwlock);
```

### 内核读写锁

```c
// include/linux/rwlock.h
rwlock_t lock;

rwlock_init(&lock);

// 读锁
read_lock(&lock);
// 读操作
read_unlock(&lock);

// 写锁
write_lock(&lock);
// 写操作
write_unlock(&lock);
```

## 八、屏障（Barrier）

### 概念

屏障用于同步多个线程到达同一点：

```c
pthread_barrier_t barrier;

// 初始化，参数是需要等待的线程数
pthread_barrier_init(&barrier, NULL, 3);

// 每个线程调用
pthread_barrier_wait(&barrier);
// 所有线程都到达后才继续

pthread_barrier_destroy(&barrier);
```

### 使用示例

```c
#define NUM_THREADS 4
pthread_barrier_t barrier;

void *thread_func(void *arg)
{
    int id = *(int *)arg;
    
    // 阶段 1
    printf("Thread %d: phase 1\n", id);
    pthread_barrier_wait(&barrier);
    
    // 阶段 2
    printf("Thread %d: phase 2\n", id);
    pthread_barrier_wait(&barrier);
    
    return NULL;
}

int main()
{
    pthread_barrier_init(&barrier, NULL, NUM_THREADS);
    
    pthread_t threads[NUM_THREADS];
    int ids[NUM_THREADS];
    
    for (int i = 0; i < NUM_THREADS; i++) {
        ids[i] = i;
        pthread_create(&threads[i], NULL, thread_func, &ids[i]);
    }
    
    for (int i = 0; i < NUM_THREADS; i++) {
        pthread_join(threads[i], NULL);
    }
    
    pthread_barrier_destroy(&barrier);
    return 0;
}
```

## 九、同步机制对比

### 选择建议

| 场景 | 推荐机制 |
|-----|---------|
| 短时间临界区 | 自旋锁 |
| 长时间临界区 | 互斥锁 |
| 多读少写 | 读写锁 |
| 生产者-消费者 | 条件变量 |
| 同步点 | 屏障 |

### 性能对比

| 机制 | 无竞争开销 | 有竞争开销 | CPU 使用 |
|-----|-----------|-----------|---------|
| 自旋锁 | 极低 | 高（忙等待） | 高 |
| 互斥锁 | 低 | 中（可能睡眠） | 低 |
| 读写锁 | 中 | 中 | 中 |

## 总结

### 核心要点

1. **Linux 线程模型**：
   - NPTL，1:1 模型
   - 基于 clone() 实现
   - 线程是共享资源的 task_struct

2. **同步机制**：
   - 互斥锁：通用同步
   - 条件变量：等待条件
   - 自旋锁：短临界区
   - 读写锁：多读少写
   - 屏障：同步点

3. **Futex**：
   - 快速用户空间互斥
   - 无竞争时不进内核
