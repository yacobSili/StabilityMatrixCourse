# 多核调度与负载均衡

## 学习目标

- 理解调度域（sched_domain）和调度组的概念
- 掌握多核调度机制
- 理解负载均衡算法
- 了解 CPU 亲和性（affinity）
- 理解能效感知调度（EAS）和 Android 大小核策略

## 概述

在多核系统中，调度器需要解决：
1. 如何在多个 CPU 之间分配任务
2. 如何均衡各 CPU 的负载
3. 如何考虑 CPU 拓扑结构（缓存、NUMA）
4. 如何平衡性能和功耗

## 一、调度域（Scheduler Domain）

### 概念

调度域表示一组可以进行负载均衡的 CPU 集合，反映了系统的拓扑结构：

```
典型的调度域层次（大小核系统）：

              ┌─────────────────────────────────┐
              │        DIE Domain               │ ← 整个芯片
              │   (所有 CPU 共享)               │
              └─────────────────┬───────────────┘
                                │
        ┌───────────────────────┴───────────────────────┐
        │                                               │
┌───────▼───────┐                               ┌───────▼───────┐
│  MC Domain    │ ← 共享 L2 缓存                │  MC Domain    │
│  (大核集群)   │                               │  (小核集群)   │
└───────┬───────┘                               └───────┬───────┘
        │                                               │
┌───────▼───────┐                               ┌───────▼───────┐
│  SMT Domain   │ ← 超线程（如果有）             │  SMT Domain   │
└───────────────┘                               └───────────────┘
```

### load_balance() - 负载均衡核心

```c
// kernel/sched/fair.c
static int load_balance(int this_cpu, struct rq *this_rq,
                        struct sched_domain *sd, enum cpu_idle_type idle,
                        int *continue_balancing)
{
    int ld_moved;
    struct lb_env env = {
        .sd             = sd,
        .dst_cpu        = this_cpu,
        .dst_rq         = this_rq,
        .idle           = idle,
        .loop_max       = sysctl_sched_nr_migrate,
    };
    
    // 1. 找到最繁忙的组
    struct sched_group *busiest = find_busiest_group(&env);
    if (!busiest)
        goto out_balanced;
    
    // 2. 找到最繁忙的 CPU
    env.src_rq = find_busiest_queue(&env, busiest);
    if (!env.src_rq)
        goto out_balanced;
    
    // 3. 迁移任务
    ld_moved = 0;
    if (env.src_rq->nr_running > 1) {
        env.src_cpu = cpu_of(env.src_rq);
        env.loop = 0;
        
        // 锁定源运行队列
        raw_spin_lock_irqsave(&env.src_rq->lock, flags);
        
        // 执行迁移
        ld_moved = detach_tasks(&env);
        raw_spin_unlock(&env.src_rq->lock);
        
        if (ld_moved) {
            attach_tasks(&env);
        }
    }
    
    // 4. 更新统计
    sd->nr_balance_failed = 0;
    
    return ld_moved;

out_balanced:
    return 0;
}
```

```c
};

// 更新负载
static inline void update_load_avg(struct cfs_rq *cfs_rq, struct sched_entity *se,
                                   int flags)
{
    u64 now = cfs_rq_clock_task(cfs_rq);
    
    // 更新实体的负载平均
    if (se->avg.last_update_time && !(flags & SKIP_AGE_LOAD))
        __update_load_avg_se(now, cfs_rq, se);
    
    // 更新 cfs_rq 的负载平均
    update_cfs_rq_load_avg(now, cfs_rq);
}
```

### 迁移决策

```c
// kernel/sched/fair.c
static struct sched_group *find_busiest_group(struct lb_env *env)
{
    struct sd_lb_stats sds;
    
    // 1. 更新所有组的统计
    update_sd_lb_stats(env, &sds);
    
    // 2. 检查是否需要平衡
    if (!sds.busiest || sds.busiest_nr_running == 0)
        goto out_balanced;
    
    // 3. 计算不平衡量
    calculate_imbalance(env, &sds);
    
    if (sds.avg_load < sds.busiest_load) {
        // 需要从 busiest 迁移任务
        return sds.busiest;
    }

out_balanced:
    return NULL;
}
```

## 三、CPU 选择

### select_task_rq_fair() - 选择目标 CPU

```c
// kernel/sched/fair.c
static int select_task_rq_fair(struct task_struct *p, int prev_cpu,
                               int sd_flag, int wake_flags)
{
    struct sched_domain *sd;
    int new_cpu = prev_cpu;
    int want_affine = 0;
    
    // 1. 检查亲和性
    if (sd_flag & SD_BALANCE_WAKE) {
        want_affine = !wake_wide(p) && 
                      cpumask_test_cpu(cpu, p->cpus_ptr);
    }
    
    // 2. 遍历调度域
    for_each_domain(cpu, sd) {
        if (sd->flags & SD_WAKE_AFFINE) {
            // 亲和性唤醒
            new_cpu = select_idle_sibling(p, prev_cpu, new_cpu);
            break;
        }
    }
    
    // 3. EAS 选择（如果启用）
    if (sched_energy_enabled()) {
        new_cpu = find_energy_efficient_cpu(p, prev_cpu);
        if (new_cpu >= 0)
            return new_cpu;
    }
    
    return new_cpu;
}
```

### select_idle_sibling() - 选择空闲的兄弟 CPU

```c
// kernel/sched/fair.c
static int select_idle_sibling(struct task_struct *p, int prev, int target)
{
    struct sched_domain *sd;
    int i;
    
    // 1. 检查目标 CPU 是否空闲
    if (available_idle_cpu(target))
        return target;
    
    // 2. 检查前一个 CPU 是否空闲
    if (prev != target && available_idle_cpu(prev))
        return prev;
    
    // 3. 搜索空闲 CPU
    sd = rcu_dereference(per_cpu(sd_llc, target));
    if (sd) {
        // 在共享 LLC 的 CPU 中搜索
        for_each_cpu_wrap(i, sched_domain_span(sd), target) {
            if (available_idle_cpu(i))
                return i;
        }
    }
    
    return target;
}
```

## 四、CPU 亲和性

### 概念

CPU 亲和性限制进程可以运行的 CPU 集合。

### 设置亲和性

```c
// 系统调用
int sched_setaffinity(pid_t pid, size_t cpusetsize, const cpu_set_t *mask);
int sched_getaffinity(pid_t pid, size_t cpusetsize, cpu_set_t *mask);

// 使用示例
cpu_set_t mask;
CPU_ZERO(&mask);
CPU_SET(0, &mask);  // 只允许在 CPU 0 运行
CPU_SET(1, &mask);  // 也允许在 CPU 1 运行
sched_setaffinity(pid, sizeof(mask), &mask);
```

### 内核实现

```c
// kernel/sched/core.c
SYSCALL_DEFINE3(sched_setaffinity, pid_t, pid, unsigned int, len,
                unsigned long __user *, user_mask_ptr)
{
    cpumask_var_t new_mask;
    int retval;
    
    // 复制用户空间的掩码
    if (copy_from_user(new_mask, user_mask_ptr, len))
        return -EFAULT;
    
    // 设置亲和性
    retval = sched_setaffinity(pid, new_mask);
    
    return retval;
}

static int __sched_setaffinity(struct task_struct *p, const struct cpumask *mask)
{
    int retval;
    
    // 检查掩码是否有效
    if (!cpumask_intersects(mask, cpu_online_mask))
        return -EINVAL;
    
    // 更新 cpus_mask
    cpumask_copy(&p->cpus_mask, mask);
    p->nr_cpus_allowed = cpumask_weight(mask);
    
    // 如果当前 CPU 不在新掩码中，迁移
    if (!cpumask_test_cpu(task_cpu(p), mask))
        migrate_task_to(p, cpumask_any_and(mask, cpu_online_mask));
    
    return 0;
}
```

## 五、能效感知调度（EAS）

### 概念

EAS（Energy Aware Scheduling）在调度决策中考虑能耗，在满足性能需求的前提下最小化功耗。

### 大小核架构

```
典型的 ARM big.LITTLE 架构：

┌─────────────────────────────────────────────────────────────┐
│                         SoC                                  │
│  ┌─────────────────────┐     ┌─────────────────────────┐   │
│  │    大核集群          │     │      小核集群            │   │
│  │  (高性能，高功耗)    │     │   (低性能，低功耗)       │   │
│  │  ┌───┐ ┌───┐ ┌───┐ │     │   ┌───┐ ┌───┐ ┌───┐    │   │
│  │  │C0 │ │C1 │ │C2 │ │     │   │C4 │ │C5 │ │C6 │    │   │
│  │  └───┘ └───┘ └───┘ │     │   └───┘ └───┘ └───┘    │   │
│  │      共享 L2        │     │      共享 L2            │   │
│  └─────────────────────┘     └─────────────────────────┘   │
│                    共享 L3                                  │
└─────────────────────────────────────────────────────────────┘
```

### CPU 容量

### find_energy_efficient_cpu() - 选择能效最优 CPU

```c
// kernel/sched/fair.c
static int find_energy_efficient_cpu(struct task_struct *p, int prev_cpu)
{
    unsigned long prev_delta = ULONG_MAX;
    unsigned long best_delta = ULONG_MAX;
    int best_cpu = -1;
    
    // 遍历性能域
    for_each_performance_domain(pd) {
        unsigned long base_energy;
        
        // 计算当前能耗
        base_energy = compute_energy(p, prev_cpu, pd);
        
        // 尝试每个 CPU
        for_each_cpu(cpu, cpumask_of_pd(pd)) {
            if (!cpumask_test_cpu(cpu, p->cpus_ptr))
                continue;
            
            // 计算如果放到这个 CPU 的能耗
            unsigned long new_energy = compute_energy(p, cpu, pd);
            
            // 选择能耗最低的
            if (new_energy < best_delta) {
                best_delta = new_energy;
                best_cpu = cpu;
            }
        }
    }
    
    return best_cpu;
}
```

### 能耗模型

```c
// 能耗 = 动态功耗 + 静态功耗
// 动态功耗 = C × V² × f × utilization
// 其中：C = 电容，V = 电压，f = 频率

// kernel/sched/fair.c
static unsigned long compute_energy(struct task_struct *p, int dst_cpu,
                                    struct perf_domain *pd)
{
    struct cpumask *pd_mask = cpumask_of_pd(pd);
    unsigned long energy = 0;
    int cpu;
    
    for_each_cpu(cpu, pd_mask) {
        unsigned long util = cpu_util_next(cpu, p, dst_cpu);
        unsigned long cpu_cap = capacity_of(cpu);
        
        // 查能耗表
        energy += em_cpu_energy(em, cpu_cap, util);
    }
    
    return energy;
}
```

## 六、Android 大小核策略

### schedtune/cpuset

```bash
# Android 使用 cgroup 控制进程的 CPU 亲和性

# 创建 cpuset
/dev/cpuset/
├── foreground/      # 前台进程
│   └── cpus         # 所有 CPU
├── background/      # 后台进程
│   └── cpus         # 只用小核
├── top-app/         # 顶层应用
│   └── cpus         # 优先大核
└── system-background/
    └── cpus         # 系统后台
```

### 配置示例

```bash
# 前台应用可以使用所有 CPU
echo 0-7 > /dev/cpuset/foreground/cpus

# 后台应用只能使用小核
echo 0-3 > /dev/cpuset/background/cpus

# 顶层应用优先使用大核
echo 4-7 > /dev/cpuset/top-app/cpus

# 将进程加入 cpuset
echo <pid> > /dev/cpuset/foreground/tasks
```

### UCLAMP（Utilization Clamping）

```c
// 限制进程的利用率范围
// 防止低优先级任务使用过多 CPU

// 设置利用率上限

sched_setattr(pid, &attr, 0);
```

## 总结

### 核心要点

1. **调度域**：
   - 反映 CPU 拓扑结构
   - 层次化组织（SMT → MC → DIE）
   - 影响负载均衡范围

2. **负载均衡**：
   - 定期触发（tick）
   - CPU 空闲时触发
   - 进程唤醒/fork/exec 时触发

3. **CPU 亲和性**：
   - 限制进程可运行的 CPU
   - 通过 sched_setaffinity 设置

4. **能效感知调度（EAS）**：
   - 考虑 CPU 能耗
   - 大小核调度
   - Android 广泛使用

5. **Android 策略**：
   - cpuset 控制 CPU 亲和性
   - 前台/后台差异化调度
   - UCLAMP 限制利用率
