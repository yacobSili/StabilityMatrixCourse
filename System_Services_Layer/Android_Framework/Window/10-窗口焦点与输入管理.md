# 窗口焦点与输入管理

## 引言

窗口焦点决定了输入事件的分发目标。WMS 需要维护焦点窗口状态，并与 InputManagerService 协作完成输入分发。本文将详细介绍焦点管理机制以及与 Input 系统的交互。

---

## 1. 焦点类型与概念

### 1.1 焦点类型

```

│                        焦点类型                                      │
├─────────────────────────────────────────────────────────────────────┤

│     • 定义：触摸事件的目标窗口                                        │

│     • 特点：根据触摸位置动态确定                                      │

│     • 定义：前台应用/Activity                                        │
│     • 管理者：ATMS (ActivityTaskManagerService)                     │
```

## 2. 焦点计算与更新

### 2.1 updateFocusedWindowLocked

```java
// WindowManagerService.java
boolean updateFocusedWindowLocked(int mode, boolean updateInputWindows) {
    // 计算新的焦点窗口
    WindowState newFocus = computeFocusedWindowLocked();
    
    if (mCurrentFocus != newFocus) {
        // 焦点变化
        mH.removeMessages(H.REPORT_FOCUS_CHANGE);
        mH.sendEmptyMessage(H.REPORT_FOCUS_CHANGE);
        
        final WindowState oldFocus = mCurrentFocus;
        mCurrentFocus = newFocus;
        mLastFocus = oldFocus;
        
        // 通知 ATMS 焦点变化
        if (newFocus != null) {
            mAtmService.setResumedActivityUncheckLocked(
                    newFocus.mActivityRecord, "updateFocusedWindowLocked");
        }
        
        // 同步到 InputDispatcher
        if (updateInputWindows) {
            mInputMonitor.setInputFocusLw(newFocus, updateInputWindows);
        }
        
        return true;
    }
    return false;
}
```

### 2.2 computeFocusedWindowLocked

```java
// DisplayContent.java
WindowState findFocusedWindow() {
    // 自顶向下遍历窗口，找到第一个可聚焦的
    return getWindow(w -> {
        // 1. 检查 Token 是否允许焦点
        if (!w.mToken.windowsAreFocusable()) {
            return false;
        }
        
        // 2. 检查窗口本身是否可聚焦
        if (!w.canReceiveKeys()) {
            return false;
        }
        
        // 3. 找到了焦点窗口
        return true;
    });
}

// WindowState.java
boolean canReceiveKeys() {
    return isVisibleOrAdding()
            && !mRemoved
            && !mRemoveOnExit
            // FLAG_NOT_FOCUSABLE 标志
            && (mAttrs.flags & WindowManager.LayoutParams.FLAG_NOT_FOCUSABLE) == 0
            // Token 允许焦点
            && mToken.windowsAreFocusable()
            // 不在退出动画中
            && !mAnimatingExit;
}

// WindowToken.java
boolean windowsAreFocusable() {
    // ActivityRecord 的情况
    if (asActivityRecord() != null) {
        return asActivityRecord().windowsAreFocusable();
    }
    return true;
}

// ActivityRecord.java
boolean windowsAreFocusable() {
    // 检查 Activity 是否可聚焦
    return !mHidden
            && mVisibleRequested
            && !mActivityComponent.equals(mAtmService.getHomeActivityComponent());
}
```

## 3. 焦点变化通知

### 3.1 通知应用

```java
// WindowManagerService.java
class H extends Handler {
    @Override
    public void handleMessage(Message msg) {
        switch (msg.what) {
            case REPORT_FOCUS_CHANGE: {
                WindowState lastFocus;
                WindowState newFocus;
                
                synchronized (mGlobalLock) {
                    lastFocus = mLastFocus;
                    newFocus = mCurrentFocus;
                    
                    // 避免重复通知
                    if (lastFocus == newFocus) {
                        return;
                    }
                }
                
                // 通知失去焦点的窗口
                if (lastFocus != null) {
                    lastFocus.reportFocusChangedSerialized(false /*focused*/);
                }
                
                // 通知获得焦点的窗口
                if (newFocus != null) {
                    newFocus.reportFocusChangedSerialized(true /*focused*/);
                }
                break;
            }
        }
    }
}

// WindowState.java
void reportFocusChangedSerialized(boolean focused) {
    // 通过 IWindow 回调到应用
    mClient.windowFocusChanged(focused, mInTouchMode);
}
```

```java
void windowFocusChanged(boolean hasFocus, boolean inTouchMode) {
    // 发送消息到主线程处理
    Message msg = mHandler.obtainMessage(MSG_WINDOW_FOCUS_CHANGED);
    msg.arg1 = hasFocus ? 1 : 0;
    msg.arg2 = inTouchMode ? 1 : 0;
    mHandler.sendMessage(msg);
}

// Handler 处理
case MSG_WINDOW_FOCUS_CHANGED: {
    handleWindowFocusChanged();
    break;
}

private void handleWindowFocusChanged() {
    final boolean hasWindowFocus;
    synchronized (this) {
        hasWindowFocus = mAttachInfo.mHasWindowFocus;
    }
    
    // 通知 View 树
    if (mView != null) {
        mView.dispatchWindowFocusChanged(hasWindowFocus);
    }
    
    // 回调 Activity
    if (mAttachInfo.mContentOnApplyWindowInsetsListener != null) {
        mAttachInfo.mContentOnApplyWindowInsetsListener.onWindowFocusChanged(hasWindowFocus);
    }
}
```

## 4. 与 InputManager 的交互

### 4.1 InputMonitor

```java
// InputMonitor.java
final class InputMonitor {
    private final WindowManagerService mService;
    
    // 输入焦点窗口
    private WindowState mInputFocus;
    
    // 设置输入焦点
    void setInputFocusLw(WindowState newWindow, boolean updateInputWindows) {
        if (mInputFocus != newWindow) {
            mInputFocus = newWindow;
            
            if (updateInputWindows) {
                updateInputWindowsLw(true);
            }
        }
    }
    
    // 更新输入窗口信息到 InputDispatcher
    void updateInputWindowsLw(boolean force) {
        // 收集所有窗口的输入信息
        mUpdateInputWindowsNeeded = false;
        
        // 遍历所有窗口
        mService.mRoot.forAllWindows(mUpdateInputWindows, true);
        
        // 设置焦点应用
        setFocusedAppLw(focusedApp);
        
        // 提交到 InputManager
        mService.mInputManager.setInputWindows(mInputWindowHandles.toArray(
                new InputWindowHandle[0]), mFocusedInputWindowHandle);
    }
}
```

### 4.2 InputWindowHandle

```java
// 收集窗口输入信息
private final Consumer<WindowState> mUpdateInputWindows = w -> {
    final InputWindowHandle inputWindowHandle = w.mInputWindowHandle;
    
    if (!w.mInputChannel != null || w.mRemoved || !w.canReceiveTouchInput()) {
        return;
    }
    
    // 填充窗口信息
    inputWindowHandle.name = w.getName();
    inputWindowHandle.ownerUid = w.mOwnerUid;
    inputWindowHandle.ownerPid = w.mSession.mPid;
    inputWindowHandle.layoutParamsFlags = w.mAttrs.flags;
    inputWindowHandle.layoutParamsType = w.mAttrs.type;
    
    // 触摸区域
    inputWindowHandle.touchableRegion.set(w.mTouchableRegion);
    
    // 窗口帧
    inputWindowHandle.frameLeft = w.getFrame().left;
    inputWindowHandle.frameTop = w.getFrame().top;
    inputWindowHandle.frameRight = w.getFrame().right;
    inputWindowHandle.frameBottom = w.getFrame().bottom;
    
    // 是否可见/可触摸
    inputWindowHandle.visible = w.isVisibleLw();
    inputWindowHandle.canReceiveKeys = w.canReceiveKeys();
    inputWindowHandle.hasFocus = w == mInputFocus;
    
    mInputWindowHandles.add(inputWindowHandle);
};
```

## 5. InputChannel 管理

### 5.1 InputChannel 创建

```java
// WindowState.java
void openInputChannel(InputChannel outInputChannel) {
    if (mInputChannel != null) {
        throw new IllegalStateException("Already has input channel");
    }
    
    // 创建 InputChannel 对
    InputChannel[] inputChannels = InputChannel.openInputChannelPair(getName());
    
    // 服务端 channel
    mInputChannel = inputChannels[0];
    
    // 客户端 channel（传递给应用）
    inputChannels[1].transferTo(outInputChannel);
    
    // 注册到 InputDispatcher
    mWmService.mInputManager.registerInputChannel(mInputChannel);
}
```

## 6. 焦点策略

### 6.1 焦点策略配置

```java
// DisplayPolicy.java
void adjustWindowParamsLw(WindowState win, WindowManager.LayoutParams attrs) {
    // 根据窗口类型调整焦点行为
    switch (attrs.type) {
        case TYPE_STATUS_BAR:
        case TYPE_NAVIGATION_BAR:
            // 系统栏不获取焦点
            attrs.flags |= FLAG_NOT_FOCUSABLE;
            break;
            
        case TYPE_TOAST:
            // Toast 不获取焦点
            attrs.flags |= FLAG_NOT_FOCUSABLE | FLAG_NOT_TOUCHABLE;
            break;
            
        case TYPE_INPUT_METHOD:
            // 输入法窗口特殊处理
            attrs.flags |= FLAG_NOT_TOUCH_MODAL;
            break;
    }
}
```
