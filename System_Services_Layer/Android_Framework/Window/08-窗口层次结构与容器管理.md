# 窗口层次结构与容器管理

## 引言

Android 窗口系统采用树形层次结构来组织窗口，从 RootWindowContainer 到具体的 WindowState，形成了清晰的容器层次。本文将详细介绍这个层次结构的设计和各组件的职责。

## 2. WindowContainer 基类

### 2.1 WindowContainer 设计

```java
// WindowContainer.java
class WindowContainer<E extends WindowContainer> extends ConfigurationContainer<E>
        implements Comparable<WindowContainer>, Animatable {
    
    // 父容器
    private WindowContainer<WindowContainer> mParent;
    
    // 子容器列表
    protected final WindowList<E> mChildren = new WindowList<>();
    
    // SurfaceControl
    protected SurfaceControl mSurfaceControl;
    
    // 层级
    private int mLayer;
    
    // ======== 子容器管理 ========
    
    void addChild(E child, int index) {
        child.setParent(this);
        mChildren.add(index, child);
        onChildAdded(child);
    }
    
    void removeChild(E child) {
        mChildren.remove(child);
        child.setParent(null);
    }
    
    // ======== 遍历方法 ========
    
    // 自顶向下遍历
    boolean forAllWindows(ToBooleanFunction<WindowState> callback, boolean traverseTopToBottom) {
        if (traverseTopToBottom) {
            for (int i = mChildren.size() - 1; i >= 0; --i) {
                if (mChildren.get(i).forAllWindows(callback, traverseTopToBottom)) {
                    return true;
                }
            }
        } else {
            for (int i = 0; i < mChildren.size(); i++) {
                if (mChildren.get(i).forAllWindows(callback, traverseTopToBottom)) {
                    return true;
                }
            }
        }
        return false;
    }
    
    // 获取顶部窗口
    WindowState getTopChild() {
        for (int i = mChildren.size() - 1; i >= 0; --i) {
            final WindowState w = mChildren.get(i).getTopChild();
            if (w != null) return w;
        }
        return null;
    }
}
```

### 2.2 配置继承

```java
// ConfigurationContainer.java
abstract class ConfigurationContainer<E extends ConfigurationContainer> {
    
    // 请求的配置
    private Configuration mRequestedOverrideConfiguration;
    
    // 完整配置 = 父配置 + 覆盖配置
    private Configuration mFullConfiguration;
    
    // 合并配置时的继承链
    void onConfigurationChanged(Configuration newParentConfig) {
        mFullConfiguration.setTo(newParentConfig);
        mFullConfiguration.updateFrom(mRequestedOverrideConfiguration);
        
        // 通知子容器
        for (int i = mChildren.size() - 1; i >= 0; --i) {
            mChildren.get(i).onConfigurationChanged(mFullConfiguration);
        }
    }
}
```

## 3. RootWindowContainer

### 3.1 RootWindowContainer 职责

```java
// RootWindowContainer.java
class RootWindowContainer extends WindowContainer<DisplayContent>
        implements DisplayManager.DisplayListener {
    
    // WMS 引用
    final WindowManagerService mWmService;
    
    // 所有 DisplayContent
    private final SparseArray<DisplayContent> mDisplayContents = new SparseArray<>();
    
    // ======== 显示管理 ========
    
    // 获取或创建 DisplayContent
    DisplayContent getDisplayContentOrCreate(int displayId) {
        DisplayContent dc = mDisplayContents.get(displayId);
        if (dc == null) {
            final Display display = mDisplayManager.getDisplay(displayId);
            if (display != null) {
                dc = createDisplayContent(display);
            }
        }
        return dc;
    }
    
    // 创建 DisplayContent
    DisplayContent createDisplayContent(Display display) {
        DisplayContent dc = new DisplayContent(display, mWmService);
        addChild(dc, POSITION_TOP);
        return dc;
    }
    
    // ======== 焦点管理 ========
    
    // 获取焦点窗口
    WindowState getCurrentInputMethodWindow() {
        // 遍历所有 Display 找到 IME 窗口
    }
    
    // ======== 布局计算 ========
    
    void performSurfacePlacement() {
        // 遍历所有 DisplayContent 执行布局
        forAllDisplays(dc -> {
            dc.performLayout(false, false);
            dc.prepareSurfaces();
        });
    }
}
```

## 4. DisplayContent

### 4.1 DisplayContent 结构

```java
// DisplayContent.java
class DisplayContent extends WindowContainer<DisplayArea>
        implements WindowManagerPolicy.DisplayContentInfo {
    
    // 显示 ID
    private final int mDisplayId;
    
    // 显示信息
    final DisplayInfo mDisplayInfo = new DisplayInfo();
    
    // 显示策略
    private DisplayPolicy mDisplayPolicy;
    
    // 输入监控
    final InputMonitor mInputMonitor;
    
    // 任务显示区域
    private TaskDisplayArea mTaskDisplayArea;
    
    // 输入法容器
    private DisplayArea.Tokens mImeContainer;
    
    // 当前焦点窗口
    WindowState mCurrentFocus;
    WindowState mLastFocus;
    
    // 壁纸控制器
    final WallpaperController mWallpaperController;
    
    // ======== 核心方法 ========
    
    // 获取焦点窗口
    WindowState findFocusedWindow() {
        // 自顶向下遍历，找到第一个可聚焦的窗口
        return getWindow(w -> {
            if (w.canReceiveKeys()) {
                return true;
            }
            return false;
        });
    }
    
    // 执行布局
    void performLayout(boolean initial, boolean updateInputWindows) {
        // 1. 开始布局
        mDisplayPolicy.beginLayoutLw(mDisplayInfo, ...);
        
        // 2. 遍历所有窗口计算布局
        forAllWindows(w -> {
            if (w.isVisible()) {
                mDisplayPolicy.layoutWindowLw(w, ...);
            }
        }, true);
        
        // 3. 更新输入窗口
        if (updateInputWindows) {
            mInputMonitor.updateInputWindowsLw(false);
        }
    }
    
    // 分配窗口层级
    void assignWindowLayers(boolean setLayoutNeeded) {
        // 调用 LayerAssigner 分配 Z-Order
        mLayersController.assignWindowLayers(mTransaction);
    }
}
```

## 5. WindowToken

### 5.1 WindowToken 职责

```java
// WindowToken.java
class WindowToken extends WindowContainer<WindowState> {
    
    // Token 标识
    final IBinder token;
    
    // 窗口类型
    final int windowType;
    
    // 是否持久化
    final boolean mPersistOnEmpty;
    
    // 所属 DisplayContent
    DisplayContent mDisplayContent;
    
    // ======== 窗口管理 ========
    
    void addWindow(final WindowState win) {
        // 按照窗口类型排序插入
        int i;
        for (i = mChildren.size() - 1; i >= 0; i--) {
            if (mChildren.get(i).mSubLayer < win.mSubLayer) {
                break;
            }
        }
        addChild(win, i + 1);
    }
    
    // ======== Token 类型 ========
    // 根据类型有不同子类：
    // - ActivityRecord: Activity 的 token
    // - WallpaperWindowToken: 壁纸 token
    // - WindowToken: 普通 token（Toast、系统窗口等）
}
```

### 5.2 ActivityRecord 作为 WindowToken

```java
// ActivityRecord.java
final class ActivityRecord extends WindowToken {
    
    // Activity 信息
    final ActivityInfo info;
    final String packageName;
    
    // 任务归属
    Task task;
    
    // 状态
    ActivityState mState = INITIALIZING;
    
    // 关联的窗口
    // WindowState 作为子节点存储在 mChildren 中
    
    // 获取主窗口
    WindowState findMainWindow() {
        return findMainWindow(true);
    }
    
    WindowState findMainWindow(boolean includeStartingApp) {
        // 查找类型为 TYPE_BASE_APPLICATION 的窗口
        WindowState candidate = null;
        for (int j = mChildren.size() - 1; j >= 0; --j) {
            final WindowState win = mChildren.get(j);
            if (win.mAttrs.type == TYPE_BASE_APPLICATION) {
                return win;
            } else if (win.mAttrs.type == TYPE_APPLICATION_STARTING) {
                if (includeStartingApp) {
                    candidate = win;
                }
            }
        }
        return candidate;
    }
}
```

## 6. WindowState

### 6.1 WindowState 核心结构

```java
// WindowState.java
class WindowState extends WindowContainer<WindowState> {
    
    // ======== 身份标识 ========
    final Session mSession;            // 客户端会话
    final IWindow mClient;             // 客户端窗口引用
    final WindowToken mToken;          // 所属 Token
    final WindowManager.LayoutParams mAttrs;  // 窗口属性
    
    // ======== 几何信息 ========
    final Rect mRequestedWidth, mRequestedHeight;
    final WindowFrames mWindowFrames;  // 窗口帧信息
    
    // ======== Surface ========
    SurfaceControl mSurfaceControl;
    final WindowSurfaceController mWinAnimator;
    
    // ======== 状态 ========
    boolean mHasSurface;              // 是否有 Surface
    boolean mRemoved;                 // 是否已移除
    boolean mDestroying;              // 是否正在销毁
    
    // ======== 输入 ========
    InputChannel mInputChannel;
    final InputWindowHandle mInputWindowHandle;
    
    // ======== 层级 ========
    int mLayer;                       // 层级
    int mSubLayer;                    // 子层级
    
    // ======== 核心方法 ========
    
    // 创建 SurfaceControl
    void createSurfaceControl(SurfaceControl.Transaction t) {
        mSurfaceControl = mWmService.makeSurfaceBuilder(mSession)
                .setName(mWinAnimator.mSurfaceTitle)
                .setContainerLayer()
                .build();
        t.show(mSurfaceControl);
    }
    
    // 获取可见性
    boolean isVisible() {
        return mHasSurface && isVisibleByPolicy() && !mAnimatingExit;
    }
    
    // 是否可以接收按键
    boolean canReceiveKeys() {
        return isVisibleOrAdding()
                && !mRemoved
                && (mAttrs.flags & FLAG_NOT_FOCUSABLE) == 0
                && mToken.windowsAreFocusable();
    }
    
    // 准备 Surface
    void prepareSurfaceLocked(SurfaceControl.Transaction t) {
        mWinAnimator.prepareSurfaceLocked(t);
    }
}
```

```java
// TaskDisplayArea.java
final class TaskDisplayArea extends DisplayArea<WindowContainer> {
    
    // 所有任务
    private final ArrayList<Task> mRootTasks = new ArrayList<>();
    
    // 启动根任务
    Task getOrCreateRootTask(int windowingMode, int activityType, ...) {
        Task task = getRootTask(windowingMode, activityType);
        if (task == null) {
            task = new Task.Builder(mAtmService)
                    .setWindowingMode(windowingMode)
                    .setActivityType(activityType)
                    .setParent(this)
                    .build();
        }
        return task;
    }

    Task getFocusedRootTask() {
        // 从顶部开始找第一个可见的任务
        for (int i = mRootTasks.size() - 1; i >= 0; --i) {
            Task task = mRootTasks.get(i);
            if (task.isTopActivityFocusable()) {
                return task;
            }
        }
        return null;
    }
}
```

## 8. 层级分配机制

### 8.1 Z-Order 分配

```java
// DisplayContent.java
void assignWindowLayers(boolean setLayoutNeeded) {
    // 获取层级分配器
    mLayerAssigner.assignWindowLayers(mTransaction);
}

// 层级分配器
class LayerAssigner {
    // 为每个 WindowContainer 分配层级
    void assignWindowLayers(SurfaceControl.Transaction t) {
        // 按照层次结构分配
        mRoot.forAllWindows(w -> {
            w.assignLayer(t, mCurrentLayer++);
        }, true);
    }
}

// WindowState 设置层级
void assignLayer(SurfaceControl.Transaction t, int layer) {
    mLayer = layer;
    if (mSurfaceControl != null) {
        t.setLayer(mSurfaceControl, layer);
    }
}
```

## 9. 本章小结

本文详细介绍了窗口层次结构：

1. **层次树**：RootWindowContainer → DisplayContent → DisplayArea → Task → ActivityRecord → WindowState
2. **WindowContainer**：基类，提供子容器管理和遍历能力
3. **DisplayContent**：管理一个显示屏的所有窗口
4. **WindowToken**：标识一组窗口，ActivityRecord 是其子类
5. **WindowState**：单个窗口的完整表示
6. **层级分配**：Z-Order 确定窗口显示顺序

下一篇文章将介绍窗口添加与布局流程。
